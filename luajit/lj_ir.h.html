221/* SLOAD mode bits, stored in op2. */
222#define IRSLOAD_PARENT 0x01 /* Coalesce with parent trace. */
223#define IRSLOAD_FRAME 0x02 /* Load 32 bits of ftsz. */
224#define IRSLOAD_TYPECHECK 0x04 /* Needs type check. */
225#define IRSLOAD_CONVERT 0x08 /* Number to integer conversion. */
226#define IRSLOAD_READONLY 0x10 /* Read-only, omit slot store. */
227#define IRSLOAD_INHERIT 0x20 /* Inherited by exits/side traces. */
228
229/* XLOAD mode, stored in op2. */
230#define IRXLOAD_READONLY 1 /* Load from read-only data. */
231#define IRXLOAD_VOLATILE 2 /* Load from volatile data. */
232#define IRXLOAD_UNALIGNED 4 /* Unaligned load. */
233
234/* BUFHDR mode, stored in op2. */
235#define IRBUFHDR_RESET 0 /* Reset buffer. */
236#define IRBUFHDR_APPEND 1 /* Append to buffer. */
237
238/* CONV mode, stored in op2. */
239#define IRCONV_SRCMASK 0x001f /* Source IRType. */
240#define IRCONV_DSTMASK 0x03e0 /* Dest. IRType (also in ir->t). */
241#define IRCONV_DSH 5
242#define IRCONV_NUM_INT ((IRT_NUM<<IRCONV_DSH)|IRT_INT)
243#define IRCONV_INT_NUM ((IRT_INT<<IRCONV_DSH)|IRT_NUM)
244#define IRCONV_SEXT 0x0800 /* Sign-extend integer to integer. */
245#define IRCONV_MODEMASK 0x0fff
246#define IRCONV_CONVMASK 0xf000
247#define IRCONV_CSH 12
248/* Number to integer conversion mode. Ordered by strength of the checks. */
249#define IRCONV_TOBIT (0<<IRCONV_CSH) /* None. Cache only: TOBIT conv. */
250#define IRCONV_ANY (1<<IRCONV_CSH) /* Any FP number is ok. */
251#define IRCONV_INDEX (2<<IRCONV_CSH) /* Check + special backprop rules. */
252#define IRCONV_CHECK (3<<IRCONV_CSH) /* Number checked for integerness. */
253
254/* TOSTR mode, stored in op2. */
255#define IRTOSTR_INT 0 /* Convert integer to string. */
256#define IRTOSTR_NUM 1 /* Convert number to string. */
257#define IRTOSTR_CHAR 2 /* Convert char value to string. */

299/* Map of itypes to non-negative numbers and their sizes. ORDER LJ_T.
300** LJ_TUPVAL/LJ_TTRACE never appear in a TValue. Use these itypes for
301** IRT_P32 and IRT_P64, which never escape the IR.
302** The various integers are only used in the IR and can only escape to
303** a TValue after implicit or explicit conversion. Their types must be
304** contiguous and next to IRT_NUM (see the typerange macros below).
305*/
306#define IRTDEF(_) \
307 _(NIL, 4) _(FALSE, 4) _(TRUE, 4) _(LIGHTUD, LJ_64 ? 8 : 4) \
308 _(STR, IRTSIZE_PGC) _(P32, 4) _(THREAD, IRTSIZE_PGC) _(PROTO, IRTSIZE_PGC) \
309 _(FUNC, IRTSIZE_PGC) _(P64, 8) _(CDATA, IRTSIZE_PGC) _(TAB, IRTSIZE_PGC) \
310 _(UDATA, IRTSIZE_PGC) \
311 _(FLOAT, 4) _(NUM, 8) _(I8, 1) _(U8, 1) _(I16, 2) _(U16, 2) \
312 _(INT, 4) _(U32, 4) _(I64, 8) _(U64, 8) \
313 _(SOFTFP, 4) /* There is room for 8 more types. */

401static LJ_AINLINE IRType itype2irt(const TValue *tv)
402{
403 if (tvisint(tv))
404 return IRT_INT;
405 else if (tvisnum(tv))
406 return IRT_NUM;
407#if LJ_64 && !LJ_GC64
408 else if (tvislightud(tv))
409 return IRT_LIGHTUD;
410#endif
411 else
412 return (IRType)~itype(tv);
413}

415static LJ_AINLINE uint32_t irt_toitype_(IRType t)
416{
417 lj_assertX(!LJ_64 || LJ_GC64 || t != IRT_LIGHTUD,
418 "no plain type tag for lightuserdata");
419 if (LJ_DUALNUM && t > IRT_NUM) {
420 return LJ_TISNUM;
421 } else {
422 lj_assertX(t <= IRT_NUM, "no plain type tag for IR type %d", t);
423 return ~(uint32_t)t;
424 }
425}

447/* Fixed references. */
448enum {
449 REF_BIAS = 0x8000,
450 REF_TRUE = REF_BIAS-3,
451 REF_FALSE = REF_BIAS-2,
452 REF_NIL = REF_BIAS-1, /* \--- Constants grow downwards. */
453 REF_BASE = REF_BIAS, /* /--- IR grows upwards. */
454 REF_FIRST = REF_BIAS+1,
455 REF_DROP = 0xffff
456};
457
458/* Note: IRMlit operands must be < REF_BIAS, too!
459** This allows for fast and uniform manipulation of all operands
460** without looking up the operand mode in lj_ir_mode:
461** - CSE calculates the maximum reference of two operands.
462** This must work with mixed reference/literal operands, too.
463** - DCE marking only checks for operand >= REF_BIAS.
464** - LOOP needs to substitute reference operands.
465** Constant references and literals must not be modified.
466*/
467
468#define IRREF2(lo, hi) ((IRRef2)(lo) | ((IRRef2)(hi) << 16))
469
470#define irref_isk(ref) ((ref) < REF_BIAS)

472/* Tagged IR references (32 bit).
473**
474** +-------+-------+---------------+
475** | irt | flags | ref |
476** +-------+-------+---------------+
477**
478** The tag holds a copy of the IRType and speeds up IR type checks.
479*/
480typedef uint32_t TRef;
481
482#define TREF_REFMASK 0x0000ffff
483#define TREF_FRAME 0x00010000
484#define TREF_CONT 0x00020000
485
486#define TREF(ref, t) ((TRef)((ref) + ((t)<<24)))
487
488#define tref_ref(tr) ((IRRef1)(tr))
489#define tref_t(tr) ((IRType)((tr)>>24))
490#define tref_type(tr) ((IRType)(((tr)>>24) & IRT_TYPE))
491#define tref_typerange(tr, first, last) \
492 ((((tr)>>24) & IRT_TYPE) - (TRef)(first) <= (TRef)(last-first))
493
494#define tref_istype(tr, t) (((tr) & (IRT_TYPE<<24)) == ((t)<<24))
495#define tref_isnil(tr) (tref_istype((tr), IRT_NIL))
496#define tref_isfalse(tr) (tref_istype((tr), IRT_FALSE))
497#define tref_istrue(tr) (tref_istype((tr), IRT_TRUE))
498#define tref_islightud(tr) (tref_istype((tr), IRT_LIGHTUD))
499#define tref_isstr(tr) (tref_istype((tr), IRT_STR))
500#define tref_isfunc(tr) (tref_istype((tr), IRT_FUNC))
501#define tref_iscdata(tr) (tref_istype((tr), IRT_CDATA))
502#define tref_istab(tr) (tref_istype((tr), IRT_TAB))
503#define tref_isudata(tr) (tref_istype((tr), IRT_UDATA))
504#define tref_isnum(tr) (tref_istype((tr), IRT_NUM))
505#define tref_isint(tr) (tref_istype((tr), IRT_INT))
506
507#define tref_isbool(tr) (tref_typerange((tr), IRT_FALSE, IRT_TRUE))
508#define tref_ispri(tr) (tref_typerange((tr), IRT_NIL, IRT_TRUE))
509#define tref_istruecond(tr) (!tref_typerange((tr), IRT_NIL, IRT_FALSE))
510#define tref_isinteger(tr) (tref_typerange((tr), IRT_I8, IRT_INT))
511#define tref_isnumber(tr) (tref_typerange((tr), IRT_NUM, IRT_INT))
512#define tref_isnumber_str(tr) (tref_isnumber((tr)) || tref_isstr((tr)))
513#define tref_isgcv(tr) (tref_typerange((tr), IRT_STR, IRT_UDATA))

515#define tref_isk(tr) (irref_isk(tref_ref((tr))))

525/* IR instruction format (64 bit).
526**
527** 16 16 8 8 8 8
528** +-------+-------+---+---+---+---+
529** | op1 | op2 | t | o | r | s |
530** +-------+-------+---+---+---+---+
531** | op12/i/gco32 | ot | prev | (alternative fields in union)
532** +-------+-------+---+---+---+---+
533** | TValue/gco64 | (2nd IR slot for 64 bit constants)
534** +---------------+-------+-------+
535** 32 16 16
536**
537** prev is only valid prior to register allocation and then reused for r + s.
538*/
539
540typedef union IRIns {
541 struct {
542 LJ_ENDIAN_LOHI(
543 IRRef1 op1; /* IR operand 1. */
544 , IRRef1 op2; /* IR operand 2. */
545 )
546 IROpT ot; /* IR opcode and type (overlaps t and o). */
547 IRRef1 prev; /* Previous ins in same chain (overlaps r and s). */
548 };
549 struct {
550 IRRef2 op12; /* IR operand 1 and 2 (overlaps op1 and op2). */
551 LJ_ENDIAN_LOHI(
552 IRType1 t; /* IR type. */
553 , IROp1 o; /* IR opcode. */
554 )
555 LJ_ENDIAN_LOHI(
556 uint8_t r; /* Register allocation (overlaps prev). */
557 , uint8_t s; /* Spill slot allocation (overlaps prev). */
558 )
559 };
560 int32_t i; /* 32 bit signed integer literal (overlaps op12). */
561 GCRef gcr; /* GCobj constant (overlaps op12 or entire slot). */
562 MRef ptr; /* Pointer constant (overlaps op12 or entire slot). */
563 TValue tv; /* TValue constant (overlaps entire slot). */
564} IRIns;
565
566#define ir_isk64(ir) \
567 ((ir)->o == IR_KNUM || (ir)->o == IR_KINT64 || \
568 (LJ_GC64 && \
569 ((ir)->o == IR_KGC || (ir)->o == IR_KPTR || (ir)->o == IR_KKPTR)))
570
571#define ir_kgc(ir) check_exp((ir)->o == IR_KGC, gcref((ir)[LJ_GC64].gcr))
572#define ir_kstr(ir) (gco2str(ir_kgc((ir))))
573#define ir_ktab(ir) (gco2tab(ir_kgc((ir))))
574#define ir_kfunc(ir) (gco2func(ir_kgc((ir))))
575#define ir_kcdata(ir) (gco2cd(ir_kgc((ir))))
576#define ir_knum(ir) check_exp((ir)->o == IR_KNUM, &(ir)[1].tv)
577#define ir_kint64(ir) check_exp((ir)->o == IR_KINT64, &(ir)[1].tv)
578#define ir_k64(ir) check_exp(ir_isk64(ir), &(ir)[1].tv)
579#define ir_kptr(ir) \
580 check_exp((ir)->o == IR_KPTR || (ir)->o == IR_KKPTR, \
581 mref((ir)[LJ_GC64].ptr, void))

583/* A store or any other op with a non-weak guard has a side-effect. */
584static LJ_AINLINE int ir_sideeff(IRIns *ir)
585{
586 return (((ir->t.irt | ~IRT_GUARD) & lj_ir_mode[ir->o]) >= IRM_S);
587}
588
589LJ_STATIC_ASSERT((int)IRT_GUARD == (int)IRM_W);
590
591/* Replace IR instruction with NOP. */
592static LJ_AINLINE void lj_ir_nop(IRIns *ir)
593{
594 ir->ot = IRT(IR_NOP, IRT_NIL);
595 ir->op1 = ir->op2 = 0;
596 ir->prev = 0;
597}
