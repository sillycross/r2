19static void buf_grow(SBuf *sb, MSize sz)
20{
21 MSize osz = sbufsz(sb), len = sbuflen(sb), nsz = osz;
22 char *b;
23 GCSize flag;
24 if (nsz < LJ_MIN_SBUF) nsz = LJ_MIN_SBUF;
25 while (nsz < sz) nsz += nsz;
26 flag = sbufflag(sb);
27 if ((flag & SBUF_FLAG_COW)) { /* Copy-on-write semantics. */
28 lj_assertG_(G(sbufL(sb)), sb->w == sb->e, "bad SBuf COW");
29 b = (char *)lj_mem_new(sbufL(sb), nsz);
30 setsbufflag(sb, flag & ~(GCSize)SBUF_FLAG_COW);
31 setgcrefnull(sbufX(sb)->cowref);
32 memcpy(b, sb->b, osz);
33 } else {
34 b = (char *)lj_mem_realloc(sbufL(sb), sb->b, osz, nsz);
35 }
36 if ((flag & SBUF_FLAG_EXT)) {
37 sbufX(sb)->r = sbufX(sb)->r - sb->b + b; /* Adjust read pointer, too. */
38 }
39 /* Adjust buffer pointers. */
40 sb->b = b;
41 sb->w = b + len;
42 sb->e = b + nsz;
43 if ((flag & SBUF_FLAG_BORROW)) { /* Adjust borrowed buffer pointers. */
44 SBuf *bsb = mref(sbufX(sb)->bsb, SBuf);
45 bsb->b = b;
46 bsb->w = b + len;
47 bsb->e = b + nsz;
48 }
49}

51LJ_NOINLINE char *LJ_FASTCALL lj_buf_need2(SBuf *sb, MSize sz)
52{
53 lj_assertG_(G(sbufL(sb)), sz > sbufsz(sb), "SBuf overflow");
54 if (LJ_UNLIKELY(sz > LJ_MAX_BUF))
55 lj_err_mem(sbufL(sb));
56 buf_grow(sb, sz);
57 return sb->b;
58}

60LJ_NOINLINE char *LJ_FASTCALL lj_buf_more2(SBuf *sb, MSize sz)
61{
62 if (sbufisext(sb)) {
63 SBufExt *sbx = (SBufExt *)sb;
64 MSize len = sbufxlen(sbx);
65 if (LJ_UNLIKELY(sz > LJ_MAX_BUF || len + sz > LJ_MAX_BUF))
66 lj_err_mem(sbufL(sbx));
67 if (len + sz > sbufsz(sbx)) { /* Must grow. */
68 buf_grow((SBuf *)sbx, len + sz);
69 } else if (sbufxslack(sbx) < (sbufsz(sbx) >> 3)) {
70 /* Also grow to avoid excessive compactions, if slack < size/8. */
71 buf_grow((SBuf *)sbx, sbuflen(sbx) + sz); /* Not sbufxlen! */
72 return sbx->w;
73 }
74 if (sbx->r != sbx->b) { /* Compact by moving down. */
75 memmove(sbx->b, sbx->r, len);
76 sbx->r = sbx->b;
77 sbx->w = sbx->b + len;
78 lj_assertG_(G(sbufL(sbx)), len + sz <= sbufsz(sbx), "bad SBuf compact");
79 }
80 } else {
81 MSize len = sbuflen(sb);
82 lj_assertG_(G(sbufL(sb)), sz > sbufleft(sb), "SBuf overflow");
83 if (LJ_UNLIKELY(sz > LJ_MAX_BUF || len + sz > LJ_MAX_BUF))
84 lj_err_mem(sbufL(sb));
85 buf_grow(sb, len + sz);
86 }
87 return sb->w;
88}

90void LJ_FASTCALL lj_buf_shrink(lua_State *L, SBuf *sb)
91{
92 char *b = sb->b;
93 MSize osz = (MSize)(sb->e - b);
94 if (osz > 2*LJ_MIN_SBUF) {
95 MSize n = (MSize)(sb->w - b);
96 b = lj_mem_realloc(L, b, osz, (osz >> 1));
97 sb->b = b;
98 sb->w = b + n;
99 sb->e = b + (osz >> 1);
100 }
101 lj_assertG_(G(sbufL(sb)), !sbufisext(sb), "YAGNI shrink SBufExt");
102}

104char * LJ_FASTCALL lj_buf_tmp(lua_State *L, MSize sz)
105{
106 SBuf *sb = &G(L)->tmpbuf;
107 setsbufL(sb, L);
108 return lj_buf_need(sb, sz);
109}

113SBuf *lj_buf_putmem(SBuf *sb, const void *q, MSize len)
114{
115 char *w = lj_buf_more(sb, len);
116 w = lj_buf_wmem(w, q, len);
117 sb->w = w;
118 return sb;
119}

121SBuf * LJ_FASTCALL lj_buf_putchar(SBuf *sb, int c)
122{
123 char *w = lj_buf_more(sb, 1);
124 *w++ = (char)c;
125 sb->w = w;
126 return sb;
127}

129SBuf * LJ_FASTCALL lj_buf_putstr(SBuf *sb, GCstr *s)
130{
131 MSize len = s->len;
132 char *w = lj_buf_more(sb, len);
133 w = lj_buf_wmem(w, strdata(s), len);
134 sb->w = w;
135 return sb;

140SBuf * LJ_FASTCALL lj_buf_putstr_reverse(SBuf *sb, GCstr *s)
141{
142 MSize len = s->len;
143 char *w = lj_buf_more(sb, len), *e = w+len;
144 const char *q = strdata(s)+len-1;
145 while (w < e)
146 *w++ = *q--;
147 sb->w = w;
148 return sb;
149}

151SBuf * LJ_FASTCALL lj_buf_putstr_lower(SBuf *sb, GCstr *s)
152{
153 MSize len = s->len;
154 char *w = lj_buf_more(sb, len), *e = w+len;
155 const char *q = strdata(s);
156 for (; w < e; w++, q++) {
157 uint32_t c = *(unsigned char *)q;
158#if LJ_TARGET_PPC
159 *w = c + ((c >= 'A' && c <= 'Z') << 5);
160#else
161 if (c >= 'A' && c <= 'Z') c += 0x20;
162 *w = c;
163#endif
164 }
165 sb->w = w;
166 return sb;
167}

169SBuf * LJ_FASTCALL lj_buf_putstr_upper(SBuf *sb, GCstr *s)
170{
171 MSize len = s->len;
172 char *w = lj_buf_more(sb, len), *e = w+len;
173 const char *q = strdata(s);
174 for (; w < e; w++, q++) {
175 uint32_t c = *(unsigned char *)q;
176#if LJ_TARGET_PPC
177 *w = c - ((c >= 'a' && c <= 'z') << 5);
178#else
179 if (c >= 'a' && c <= 'z') c -= 0x20;
180 *w = c;
181#endif
182 }
183 sb->w = w;
184 return sb;
185}

187SBuf *lj_buf_putstr_rep(SBuf *sb, GCstr *s, int32_t rep)
188{
189 MSize len = s->len;
190 if (rep > 0 && len) {
191 uint64_t tlen = (uint64_t)rep * len;
192 char *w;
193 if (LJ_UNLIKELY(tlen > LJ_MAX_STR))
194 lj_err_mem(sbufL(sb));
195 w = lj_buf_more(sb, (MSize)tlen);
196 if (len == 1) { /* Optimize a common case. */
197 uint32_t c = strdata(s)[0];
198 do { *w++ = c; } while (--rep > 0);
199 } else {
200 const char *e = strdata(s) + len;
201 do {
202 const char *q = strdata(s);
203 do { *w++ = *q++; } while (q < e);
204 } while (--rep > 0);
205 }
206 sb->w = w;
207 }
208 return sb;
209}

211SBuf *lj_buf_puttab(SBuf *sb, GCtab *t, GCstr *sep, int32_t i, int32_t e)
212{
213 MSize seplen = sep ? sep->len : 0;
214 if (i <= e) {
215 for (;;) {
216 cTValue *o = lj_tab_getint(t, i);
217 char *w;
218 if (!o) {
219 badtype: /* Error: bad element type. */
220 sb->w = (char *)(intptr_t)i; /* Store failing index. */
221 return NULL;
222 } else if (tvisstr(o)) {
223 MSize len = strV(o)->len;
224 w = lj_buf_wmem(lj_buf_more(sb, len + seplen), strVdata(o), len);
225 } else if (tvisint(o)) {
226 w = lj_strfmt_wint(lj_buf_more(sb, STRFMT_MAXBUF_INT+seplen), intV(o));
227 } else if (tvisnum(o)) {
228 w = lj_buf_more(lj_strfmt_putfnum(sb, STRFMT_G14, numV(o)), seplen);
229 } else {
230 goto badtype;
231 }
232 if (i++ == e) {
233 sb->w = w;
234 break;
235 }
236 if (seplen) w = lj_buf_wmem(w, strdata(sep), seplen);
237 sb->w = w;
238 }
239 }
240 return sb;
241}

250/* Concatenate two strings. */
251GCstr *lj_buf_cat2str(lua_State *L, GCstr *s1, GCstr *s2)
252{
253 MSize len1 = s1->len, len2 = s2->len;
254 char *buf = lj_buf_tmp(L, len1 + len2);
255 memcpy(buf, strdata(s1), len1);
256 memcpy(buf+len1, strdata(s2), len2);
257 return lj_str_new(L, buf, len1 + len2);
258}

260/* Read ULEB128 from buffer. */
261uint32_t LJ_FASTCALL lj_buf_ruleb128(const char **pp)
262{
263 const uint8_t *w = (const uint8_t *)*pp;
264 uint32_t v = *w++;
265 if (LJ_UNLIKELY(v >= 0x80)) {
266 int sh = 0;
267 v &= 0x7f;
268 do { v |= ((*w & 0x7f) << (sh += 7)); } while (*w++ >= 0x80);
269 }
270 *pp = (const char *)w;
271 return v;
272}
