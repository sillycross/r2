36/* Synchronize data/instruction cache. */
37void lj_mcode_sync(void *start, void *end)
38{
39#ifdef LUAJIT_USE_VALGRIND
40 VALGRIND_DISCARD_TRANSLATIONS(start, (char *)end-(char *)start);
41#endif
42#if LJ_TARGET_X86ORX64
43 UNUSED(start); UNUSED(end);
44#elif LJ_TARGET_IOS
45 sys_icache_invalidate(start, (char *)end-(char *)start);
46#elif LJ_TARGET_PPC
47 lj_vm_cachesync(start, end);
48#elif defined(__GNUC__) || defined(__clang__)
49 __clear_cache(start, end);
50#else
51#error "Missing builtin to flush instruction cache"
52#endif
53}

68static void *mcode_alloc_at(jit_State *J, uintptr_t hint, size_t sz, DWORD prot)
69{
70 void *p = LJ_WIN_VALLOC((void *)hint, sz,
71 MEM_RESERVE|MEM_COMMIT|MEM_TOP_DOWN, prot);
72 if (!p && !hint)
73 lj_trace_err(J, LJ_TRERR_MCODEAL);
74 return p;
75}

77static void mcode_free(jit_State *J, void *p, size_t sz)
78{
79 UNUSED(J); UNUSED(sz);
80 VirtualFree(p, 0, MEM_RELEASE);
81}

83static int mcode_setprot(void *p, size_t sz, DWORD prot)
84{
85 DWORD oprot;
86 return !LJ_WIN_VPROTECT(p, sz, prot, &oprot);

97#define MCPROT_RW (PROT_READ|PROT_WRITE)
98#define MCPROT_RX (PROT_READ|PROT_EXEC)
99#define MCPROT_RWX (PROT_READ|PROT_WRITE|PROT_EXEC)
100#ifdef PROT_MPROTECT
101#define MCPROT_CREATE (PROT_MPROTECT(MCPROT_RWX))
102#else
103#define MCPROT_CREATE 0
104#endif

106static void *mcode_alloc_at(jit_State *J, uintptr_t hint, size_t sz, int prot)
107{
108 void *p = mmap((void *)hint, sz, prot|MCPROT_CREATE, MAP_PRIVATE|MAP_ANONYMOUS, -1, 0);
109 if (p == MAP_FAILED) {
110 if (!hint) lj_trace_err(J, LJ_TRERR_MCODEAL);
111 p = NULL;
112 }
113 return p;
114}

116static void mcode_free(jit_State *J, void *p, size_t sz)
117{
118 UNUSED(J);
119 munmap(p, sz);
120}

122static int mcode_setprot(void *p, size_t sz, int prot)
123{
124 return mprotect(p, sz, prot);

133/* -- MCode area protection ----------------------------------------------- */
134
135#if LUAJIT_SECURITY_MCODE == 0
136
137/* Define this ONLY if page protection twiddling becomes a bottleneck.
138**
139** It's generally considered to be a potential security risk to have
140** pages with simultaneous write *and* execute access in a process.
141**
142** Do not even think about using this mode for server processes or
143** apps handling untrusted external data.
144**
145** The security risk is not in LuaJIT itself -- but if an adversary finds
146** any *other* flaw in your C application logic, then any RWX memory pages
147** simplify writing an exploit considerably.
148*/
149#define MCPROT_GEN MCPROT_RWX
150#define MCPROT_RUN MCPROT_RWX
151
152static void mcode_protect(jit_State *J, int prot)
153{
154 UNUSED(J); UNUSED(prot); UNUSED(mcode_setprot);
155}
156
157#else
158
159/* This is the default behaviour and much safer:
160**
161** Most of the time the memory pages holding machine code are executable,
162** but NONE of them is writable.
163**
164** The current memory area is marked read-write (but NOT executable) only
165** during the short time window while the assembler generates machine code.
166*/
167#define MCPROT_GEN MCPROT_RW
168#define MCPROT_RUN MCPROT_RX
169
170/* Protection twiddling failed. Probably due to kernel security. */
171static LJ_NOINLINE void mcode_protfail(jit_State *J)
172{
173 lua_CFunction panic = J2G(J)->panic;
174 if (panic) {
175 lua_State *L = J->L;
176 setstrV(L, L->top++, lj_err_str(L, LJ_ERR_JITPROT));
177 panic(L);
178 }
179}
180
181/* Change protection of MCode area. */
182static void mcode_protect(jit_State *J, int prot)
183{
184 if (J->mcprot != prot) {
185 if (LJ_UNLIKELY(mcode_setprot(J->mcarea, J->szmcarea, prot)))
186 mcode_protfail(J);
187 J->mcprot = prot;
188 }
189}

195#if LJ_64
196#define mcode_validptr(p) (p)
197#else
198#define mcode_validptr(p) ((p) && (uintptr_t)(p) < 0xffff0000)
199#endif
200
201#ifdef LJ_TARGET_JUMPRANGE
202
203/* Get memory within relative jump distance of our code in 64 bit mode. */
204static void *mcode_alloc(jit_State *J, size_t sz)
205{
206 /* Target an address in the static assembler code (64K aligned).
207 ** Try addresses within a distance of target-range/2+1MB..target+range/2-1MB.
208 ** Use half the jump range so every address in the range can reach any other.
209 */
210#if LJ_TARGET_MIPS
211 /* Use the middle of the 256MB-aligned region. */
212 uintptr_t target = ((uintptr_t)(void *)lj_vm_exit_handler &
213 ~(uintptr_t)0x0fffffffu) + 0x08000000u;
214#else
215 uintptr_t target = (uintptr_t)(void *)lj_vm_exit_handler & ~(uintptr_t)0xffff;
216#endif
217 const uintptr_t range = (1u << (LJ_TARGET_JUMPRANGE-1)) - (1u << 21);
218 /* First try a contiguous area below the last one. */
219 uintptr_t hint = J->mcarea ? (uintptr_t)J->mcarea - sz : 0;
220 int i;
221 /* Limit probing iterations, depending on the available pool size. */
222 for (i = 0; i < LJ_TARGET_JUMPRANGE; i++) {
223 if (mcode_validptr(hint)) {
224 void *p = mcode_alloc_at(J, hint, sz, MCPROT_GEN);
225
226 if (mcode_validptr(p) &&
227 ((uintptr_t)p + sz - target < range || target - (uintptr_t)p < range))
228 return p;
229 if (p) mcode_free(J, p, sz); /* Free badly placed area. */
230 }
231 /* Next try probing 64K-aligned pseudo-random addresses. */
232 do {
233 hint = lj_prng_u64(&J2G(J)->prng) & ((1u<<LJ_TARGET_JUMPRANGE)-0x10000);
234 } while (!(hint + sz < range+range));
235 hint = target + hint - range;
236 }
237 lj_trace_err(J, LJ_TRERR_MCODEAL); /* Give up. OS probably ignores hints? */
238 return NULL;
239}
240
241#else
242
243/* All memory addresses are reachable by relative jumps. */
244static void *mcode_alloc(jit_State *J, size_t sz)
245{
246#if defined(__OpenBSD__) || defined(__NetBSD__) || LJ_TARGET_UWP
247 /* Allow better executable memory allocation for OpenBSD W^X mode. */
248 void *p = mcode_alloc_at(J, 0, sz, MCPROT_RUN);
249 if (p && mcode_setprot(p, sz, MCPROT_GEN)) {
250 mcode_free(J, p, sz);
251 return NULL;
252 }
253 return p;
254#else
255 return mcode_alloc_at(J, 0, sz, MCPROT_GEN);
256#endif
257}

259#endif

261/* -- MCode area management ----------------------------------------------- */
262
263/* Allocate a new MCode area. */
264static void mcode_allocarea(jit_State *J)
265{
266 MCode *oldarea = J->mcarea;
267 size_t sz = (size_t)J->param[JIT_P_sizemcode] << 10;
268 sz = (sz + LJ_PAGESIZE-1) & ~(size_t)(LJ_PAGESIZE - 1);
269 J->mcarea = (MCode *)mcode_alloc(J, sz);
270 J->szmcarea = sz;
271 J->mcprot = MCPROT_GEN;
272 J->mctop = (MCode *)((char *)J->mcarea + J->szmcarea);
273 J->mcbot = (MCode *)((char *)J->mcarea + sizeof(MCLink));
274 ((MCLink *)J->mcarea)->next = oldarea;
275 ((MCLink *)J->mcarea)->size = sz;
276 J->szallmcarea += sz;
277 J->mcbot = (MCode *)lj_err_register_mcode(J->mcarea, sz, (uint8_t *)J->mcbot);
278}
279
280/* Free all MCode areas. */
281void lj_mcode_free(jit_State *J)
282{
283 MCode *mc = J->mcarea;
284 J->mcarea = NULL;
285 J->szallmcarea = 0;
286 while (mc) {
287 MCode *next = ((MCLink *)mc)->next;
288 size_t sz = ((MCLink *)mc)->size;
289 lj_err_deregister_mcode(mc, sz, (uint8_t *)mc + sizeof(MCLink));
290 mcode_free(J, mc, sz);
291 mc = next;
292 }
293}

297/* Reserve the remainder of the current MCode area. */
298MCode *lj_mcode_reserve(jit_State *J, MCode **lim)
299{
300 if (!J->mcarea)
301 mcode_allocarea(J);
302 else
303 mcode_protect(J, MCPROT_GEN);
304 *lim = J->mcbot;
305 return J->mctop;
306}
307
308/* Commit the top part of the current MCode area. */
309void lj_mcode_commit(jit_State *J, MCode *top)
310{
311 J->mctop = top;
312 mcode_protect(J, MCPROT_RUN);
313}
314
315/* Abort the reservation. */
316void lj_mcode_abort(jit_State *J)
317{
318 if (J->mcarea)
319 mcode_protect(J, MCPROT_RUN);
320}

322/* Set/reset protection to allow patching of MCode areas. */
323MCode *lj_mcode_patch(jit_State *J, MCode *ptr, int finish)
324{
325 if (finish) {
326#if LUAJIT_SECURITY_MCODE
327 if (J->mcarea == ptr)
328 mcode_protect(J, MCPROT_RUN);
329 else if (LJ_UNLIKELY(mcode_setprot(ptr, ((MCLink *)ptr)->size, MCPROT_RUN)))
330 mcode_protfail(J);
331#endif
332 return NULL;
333 } else {
334 MCode *mc = J->mcarea;
335 /* Try current area first to use the protection cache. */
336 if (ptr >= mc && ptr < (MCode *)((char *)mc + J->szmcarea)) {
337#if LUAJIT_SECURITY_MCODE
338 mcode_protect(J, MCPROT_GEN);
339#endif
340 return mc;
341 }
342 /* Otherwise search through the list of MCode areas. */
343 for (;;) {
344 mc = ((MCLink *)mc)->next;
345 lj_assertJ(mc != NULL, "broken MCode area chain");
346 if (ptr >= mc && ptr < (MCode *)((char *)mc + ((MCLink *)mc)->size)) {
347#if LUAJIT_SECURITY_MCODE
348 if (LJ_UNLIKELY(mcode_setprot(mc, ((MCLink *)mc)->size, MCPROT_GEN)))
349 mcode_protfail(J);
350#endif
351 return mc;
352 }
353 }
354 }
355}

357/* Limit of MCode reservation reached. */
358void lj_mcode_limiterr(jit_State *J, size_t need)
359{
360 size_t sizemcode, maxmcode;
361 lj_mcode_abort(J);
362 sizemcode = (size_t)J->param[JIT_P_sizemcode] << 10;
363 sizemcode = (sizemcode + LJ_PAGESIZE-1) & ~(size_t)(LJ_PAGESIZE - 1);
364 maxmcode = (size_t)J->param[JIT_P_maxmcode] << 10;
365 if ((size_t)need > sizemcode)
366 lj_trace_err(J, LJ_TRERR_MCODEOV); /* Too long for any area. */
367 if (J->szallmcarea + sizemcode > maxmcode)
368 lj_trace_err(J, LJ_TRERR_MCODEAL);
369 mcode_allocarea(J);
370 lj_trace_err(J, LJ_TRERR_MCODELM); /* Retry with new area. */
371}
