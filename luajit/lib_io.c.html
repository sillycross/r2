29/* Userdata payload for I/O file. */
30typedef struct IOFileUD {
31 FILE *fp; /* File handle. */
32 uint32_t type; /* File type. */
33} IOFileUD;
34
35#define IOFILE_TYPE_FILE 0 /* Regular file. */
36#define IOFILE_TYPE_PIPE 1 /* Pipe. */
37#define IOFILE_TYPE_STDF 2 /* Standard file handle. */
38#define IOFILE_TYPE_MASK 3
39
40#define IOFILE_FLAG_CLOSE 4 /* Close after io.lines() iterator. */
41
42#define IOSTDF_UD(L, id) (&gcref(G(L)->gcroot[(id)])->ud)
43#define IOSTDF_IOF(L, id) ((IOFileUD *)uddata(IOSTDF_UD(L, (id))))

47static IOFileUD *io_tofilep(lua_State *L)
48{
49 if (!(L->base < L->top && tvisudata(L->base) &&
50 udataV(L->base)->udtype == UDTYPE_IO_FILE))
51 lj_err_argtype(L, 1, "FILE*");
52 return (IOFileUD *)uddata(udataV(L->base));
53}

55static IOFileUD *io_tofile(lua_State *L)
56{
57 IOFileUD *iof = io_tofilep(L);
58 if (iof->fp == NULL)
59 lj_err_caller(L, LJ_ERR_IOCLFL);
60 return iof;
61}

71static IOFileUD *io_file_new(lua_State *L)
72{
73 IOFileUD *iof = (IOFileUD *)lua_newuserdata(L, sizeof(IOFileUD));
74 GCudata *ud = udataV(L->top-1);
75 ud->udtype = UDTYPE_IO_FILE;
76 /* NOBARRIER: The GCudata is new (marked white). */
77 setgcrefr(ud->metatable, curr_func(L)->c.env);
78 iof->fp = NULL;
79 iof->type = IOFILE_TYPE_FILE;
80 return iof;
81}

83static IOFileUD *io_file_open(lua_State *L, const char *mode)
84{
85 const char *fname = strdata(lj_lib_checkstr(L, 1));
86 IOFileUD *iof = io_file_new(L);
87 iof->fp = fopen(fname, mode);
88 if (iof->fp == NULL)
89 luaL_argerror(L, 1, lj_strfmt_pushf(L, "%s: %s", fname, strerror(errno)));
90 return iof;
91}

93static int io_file_close(lua_State *L, IOFileUD *iof)
94{
95 int ok;
96 if ((iof->type & IOFILE_TYPE_MASK) == IOFILE_TYPE_FILE) {
97 ok = (fclose(iof->fp) == 0);
98 } else if ((iof->type & IOFILE_TYPE_MASK) == IOFILE_TYPE_PIPE) {
99 int stat = -1;
100#if LJ_TARGET_POSIX
101 stat = pclose(iof->fp);
102#elif LJ_TARGET_WINDOWS && !LJ_TARGET_XBOXONE && !LJ_TARGET_UWP
103 stat = _pclose(iof->fp);
104#endif
105#if LJ_52
106 iof->fp = NULL;
107 return luaL_execresult(L, stat);
108#else
109 ok = (stat != -1);
110#endif
111 } else {
112 lj_assertL((iof->type & IOFILE_TYPE_MASK) == IOFILE_TYPE_STDF,
113 "close of unknown FILE* type");
114 setnilV(L->top++);
115 lua_pushliteral(L, "cannot close standard file");
116 return 2;
117 }
118 iof->fp = NULL;
119 return luaL_fileresult(L, ok, NULL);
120}

124static int io_file_readnum(lua_State *L, FILE *fp)
125{
126 lua_Number d;
127 if (fscanf(fp, LUA_NUMBER_SCAN, &d) == 1) {
128 if (LJ_DUALNUM) {
129 int32_t i = lj_num2int(d);
130 if (d == (lua_Number)i && !tvismzero((cTValue *)&d)) {
131 setintV(L->top++, i);
132 return 1;
133 }
134 }
135 setnumV(L->top++, d);
136 return 1;
137 } else {
138 setnilV(L->top++);
139 return 0;
140 }
141}

143static int io_file_readline(lua_State *L, FILE *fp, MSize chop)
144{
145 MSize m = LUAL_BUFFERSIZE, n = 0, ok = 0;
146 char *buf;
147 for (;;) {
148 buf = lj_buf_tmp(L, m);
149 if (fgets(buf+n, m-n, fp) == NULL) break;
150 n += (MSize)strlen(buf+n);
151 ok |= n;
152 if (n && buf[n-1] == '\n') { n -= chop; break; }
153 if (n >= m - 64) m += m;
154 }
155 setstrV(L, L->top++, lj_str_new(L, buf, (size_t)n));
156 lj_gc_check(L);
157 return (int)ok;
158}

160static void io_file_readall(lua_State *L, FILE *fp)
161{
162 MSize m, n;
163 for (m = LUAL_BUFFERSIZE, n = 0; ; m += m) {
164 char *buf = lj_buf_tmp(L, m);
165 n += (MSize)fread(buf+n, 1, m-n, fp);
166 if (n != m) {
167 setstrV(L, L->top++, lj_str_new(L, buf, (size_t)n));
168 lj_gc_check(L);
169 return;
170 }
171 }
172}

174static int io_file_readlen(lua_State *L, FILE *fp, MSize m)
175{
176 if (m) {
177 char *buf = lj_buf_tmp(L, m);
178 MSize n = (MSize)fread(buf, 1, m, fp);
179 setstrV(L, L->top++, lj_str_new(L, buf, (size_t)n));
180 lj_gc_check(L);
181 return n > 0;
182 } else {
183 int c = getc(fp);
184 ungetc(c, fp);
185 setstrV(L, L->top++, &G(L)->strempty);
186 return (c != EOF);
187 }
188}

190static int io_file_read(lua_State *L, FILE *fp, int start)
191{
192 int ok, n, nargs = (int)(L->top - L->base) - start;
193 clearerr(fp);
194 if (nargs == 0) {
195 ok = io_file_readline(L, fp, 1);
196 n = start+1; /* Return 1 result. */
197 } else {
198 /* The results plus the buffers go on top of the args. */
199 luaL_checkstack(L, nargs+LUA_MINSTACK, "too many arguments");
200 ok = 1;
201 for (n = start; nargs-- && ok; n++) {
202 if (tvisstr(L->base+n)) {
203 const char *p = strVdata(L->base+n);
204 if (p[0] == '*') p++;
205 if (p[0] == 'n')
206 ok = io_file_readnum(L, fp);
207 else if ((p[0] & ~0x20) == 'L')
208 ok = io_file_readline(L, fp, (p[0] == 'l'));
209 else if (p[0] == 'a')
210 io_file_readall(L, fp);
211 else
212 lj_err_arg(L, n+1, LJ_ERR_INVFMT);
213 } else if (tvisnumber(L->base+n)) {
214 ok = io_file_readlen(L, fp, (MSize)lj_lib_checkint(L, n+1));
215 } else {
216 lj_err_arg(L, n+1, LJ_ERR_INVOPT);
217 }
218 }
219 }
220 if (ferror(fp))
221 return luaL_fileresult(L, 0, NULL);
222 if (!ok)
223 setnilV(L->top-1); /* Replace last result with nil. */
224 return n - start;
225}

227static int io_file_write(lua_State *L, FILE *fp, int start)
228{
229 cTValue *tv;
230 int status = 1;
231 for (tv = L->base+start; tv < L->top; tv++) {
232 MSize len;
233 const char *p = lj_strfmt_wstrnum(L, tv, &len);
234 if (!p)
235 lj_err_argt(L, (int)(tv - L->base) + 1, LUA_TSTRING);
236 status = status && (fwrite(p, 1, len, fp) == len);
237 }
238 if (LJ_52 && status) {
239 L->top = L->base+1;
240 if (start == 0)
241 setudataV(L, L->base, IOSTDF_UD(L, GCROOT_IO_OUTPUT));
242 return 1;
243 }
244 return luaL_fileresult(L, status, NULL);
245}

247static int io_file_iter(lua_State *L)
248{
249 GCfunc *fn = curr_func(L);
250 IOFileUD *iof = uddata(udataV(&fn->c.upvalue[0]));
251 int n = fn->c.nupvalues - 1;
252 if (iof->fp == NULL)
253 lj_err_caller(L, LJ_ERR_IOCLFL);
254 L->top = L->base;
255 if (n) { /* Copy upvalues with options to stack. */
256 lj_state_checkstack(L, (MSize)n);
257 memcpy(L->top, &fn->c.upvalue[1], n*sizeof(TValue));
258 L->top += n;
259 }
260 n = io_file_read(L, iof->fp, 0);
261 if (ferror(iof->fp))
262 lj_err_callermsg(L, strVdata(L->top-2));
263 if (tvisnil(L->base) && (iof->type & IOFILE_FLAG_CLOSE)) {
264 io_file_close(L, iof); /* Return values are ignored. */
265 return 0;
266 }
267 return n;
268}

270static int io_file_lines(lua_State *L)
271{
272 int n = (int)(L->top - L->base);
273 if (n > LJ_MAX_UPVAL)
274 lj_err_caller(L, LJ_ERR_UNPACK);
275 lua_pushcclosure(L, io_file_iter, n);
276 return 1;
277}

313LJLIB_CF(io_method_seek)
314{
315 FILE *fp = io_tofile(L)->fp;
316 int opt = lj_lib_checkopt(L, 2, 1, "\3set\3cur\3end");
317 int64_t ofs = 0;
318 cTValue *o;
319 int res;
320 if (opt == 0) opt = SEEK_SET;
321 else if (opt == 1) opt = SEEK_CUR;
322 else if (opt == 2) opt = SEEK_END;
323 o = L->base+2;
324 if (o < L->top) {
325 if (tvisint(o))
326 ofs = (int64_t)intV(o);
327 else if (tvisnum(o))
328 ofs = (int64_t)numV(o);
329 else if (!tvisnil(o))
330 lj_err_argt(L, 3, LUA_TNUMBER);
331 }
332#if LJ_TARGET_POSIX
333 res = fseeko(fp, ofs, opt);
334#elif _MSC_VER >= 1400
335 res = _fseeki64(fp, ofs, opt);
336#elif defined(__MINGW32__)
337 res = fseeko64(fp, ofs, opt);
338#else
339 res = fseek(fp, (long)ofs, opt);
340#endif
341 if (res)
342 return luaL_fileresult(L, 0, NULL);
343#if LJ_TARGET_POSIX
344 ofs = ftello(fp);
345#elif _MSC_VER >= 1400
346 ofs = _ftelli64(fp);
347#elif defined(__MINGW32__)
348 ofs = ftello64(fp);
349#else
350 ofs = (int64_t)ftell(fp);
351#endif
352 setint64V(L->top-1, ofs);
353 return 1;
354}

356LJLIB_CF(io_method_setvbuf)
357{
358 FILE *fp = io_tofile(L)->fp;
359 int opt = lj_lib_checkopt(L, 2, -1, "\4full\4line\2no");

462static int io_std_getset(lua_State *L, ptrdiff_t id, const char *mode)
463{
464 if (L->base < L->top && !tvisnil(L->base)) {
465 if (tvisudata(L->base)) {
466 io_tofile(L);
467 L->top = L->base+1;
468 } else {
469 io_file_open(L, mode);
470 }
471 /* NOBARRIER: The standard I/O handles are GC roots. */
472 setgcref(G(L)->gcroot[id], gcV(L->top-1));
473 } else {
474 setudataV(L, L->top++, IOSTDF_UD(L, id));
475 }
476 return 1;
477}

489LJLIB_CF(io_lines)
490{
491 if (L->base == L->top) setnilV(L->top++);
492 if (!tvisnil(L->base)) { /* io.lines(fname) */
493 IOFileUD *iof = io_file_open(L, "r");
494 iof->type = IOFILE_TYPE_FILE|IOFILE_FLAG_CLOSE;
495 L->top--;
496 setudataV(L, L->base, udataV(L->top));
497 } else { /* io.lines() iterates over stdin. */
498 setudataV(L, L->base, IOSTDF_UD(L, GCROOT_IO_INPUT));
499 }
500 return io_file_lines(L);
501}

503LJLIB_CF(io_type)
504{
505 cTValue *o = lj_lib_checkany(L, 1);
506 if (!(tvisudata(o) && udataV(o)->udtype == UDTYPE_IO_FILE))
507 setnilV(L->top++);
508 else if (((IOFileUD *)uddata(udataV(o)))->fp != NULL)
509 lua_pushliteral(L, "file");
510 else
511 lua_pushliteral(L, "closed file");
512 return 1;
513}

519static GCobj *io_std_new(lua_State *L, FILE *fp, const char *name)
520{
521 IOFileUD *iof = (IOFileUD *)lua_newuserdata(L, sizeof(IOFileUD));
522 GCudata *ud = udataV(L->top-1);
523 ud->udtype = UDTYPE_IO_FILE;
524 /* NOBARRIER: The GCudata is new (marked white). */
525 setgcref(ud->metatable, gcV(L->top-3));
526 iof->fp = fp;
527 iof->type = IOFILE_TYPE_STDF;
528 lua_setfield(L, -2, name);
529 return obj2gco(ud);
530}

532LUALIB_API int luaopen_io(lua_State *L)
533{
534 LJ_LIB_REG(L, NULL, io_method);
535 copyTV(L, L->top, L->top-1); L->top++;
536 lua_setfield(L, LUA_REGISTRYINDEX, LUA_FILEHANDLE);
537 LJ_LIB_REG(L, LUA_IOLIBNAME, io);
538 setgcref(G(L)->gcroot[GCROOT_IO_INPUT], io_std_new(L, stdin, "stdin"));
539 setgcref(G(L)->gcroot[GCROOT_IO_OUTPUT], io_std_new(L, stdout, "stdout"));
540 io_std_new(L, stderr, "stderr");
541 return 1;
542}
