36/* Macros to set GCobj colors and flags. */
37#define white2gray(x) ((x)->gch.marked &= (uint8_t)~LJ_GC_WHITES)
38#define gray2black(x) ((x)->gch.marked |= LJ_GC_BLACK)
39#define isfinalized(u) ((u)->marked & LJ_GC_FINALIZED)

41/* -- Mark phase ---------------------------------------------------------- */
42
43/* Mark a TValue (if needed). */
44#define gc_marktv(g, tv) \
45 { lj_assertG(!tvisgcv(tv) || (~itype(tv) == gcval(tv)->gch.gct), \
46 "TValue and GC type mismatch"); \
47 if (tviswhite(tv)) gc_mark(g, gcV(tv)); }
48
49/* Mark a GCobj (if needed). */
50#define gc_markobj(g, o) \
51 { if (iswhite(obj2gco(o))) gc_mark(g, obj2gco(o)); }
52
53/* Mark a string object. */
54#define gc_mark_str(s) ((s)->marked &= (uint8_t)~LJ_GC_WHITES)

56/* Mark a white GCobj. */
57static void gc_mark(global_State *g, GCobj *o)
58{
59 int gct = o->gch.gct;
60 lj_assertG(iswhite(o), "mark of non-white object");
61 lj_assertG(!isdead(g, o), "mark of dead object");
62 white2gray(o);
63 if (LJ_UNLIKELY(gct == ~LJ_TUDATA)) {
64 GCtab *mt = tabref(gco2ud(o)->metatable);
65 gray2black(o); /* Userdata are never gray. */
66 if (mt) gc_markobj(g, mt);
67 gc_markobj(g, tabref(gco2ud(o)->env));
68 if (LJ_HASBUFFER && gco2ud(o)->udtype == UDTYPE_BUFFER) {
69 SBufExt *sbx = (SBufExt *)uddata(gco2ud(o));
70 if (sbufiscow(sbx) && gcref(sbx->cowref))
71 gc_markobj(g, gcref(sbx->cowref));
72 if (gcref(sbx->dict))
73 gc_markobj(g, gcref(sbx->dict));
74 }
75 } else if (LJ_UNLIKELY(gct == ~LJ_TUPVAL)) {
76 GCupval *uv = gco2uv(o);
77 gc_marktv(g, uvval(uv));
78 if (uv->closed)
79 gray2black(o); /* Closed upvalues are never gray. */
80 } else if (gct != ~LJ_TSTR && gct != ~LJ_TCDATA) {
81 lj_assertG(gct == ~LJ_TFUNC || gct == ~LJ_TTAB ||
82 gct == ~LJ_TTHREAD || gct == ~LJ_TPROTO || gct == ~LJ_TTRACE,
83 "bad GC type %d", gct);
84 setgcrefr(o->gch.gclist, g->gc.gray);
85 setgcref(g->gc.gray, o);
86 }
87}

89/* Mark GC roots. */
90static void gc_mark_gcroot(global_State *g)
91{
92 ptrdiff_t i;
93 for (i = 0; i < GCROOT_MAX; i++)
94 if (gcref(g->gcroot[i]) != NULL)
95 gc_markobj(g, gcref(g->gcroot[i]));
96}

98/* Start a GC cycle and mark the root set. */
99static void gc_mark_start(global_State *g)
100{
101 setgcrefnull(g->gc.gray);
102 setgcrefnull(g->gc.grayagain);
103 setgcrefnull(g->gc.weak);
104 gc_markobj(g, mainthread(g));
105 gc_markobj(g, tabref(mainthread(g)->env));
106 gc_marktv(g, &g->registrytv);
107 gc_mark_gcroot(g);
108 g->gc.state = GCSpropagate;
109}

111/* Mark open upvalues. */
112static void gc_mark_uv(global_State *g)
113{
114 GCupval *uv;
115 for (uv = uvnext(&g->uvhead); uv != &g->uvhead; uv = uvnext(uv)) {
116 lj_assertG(uvprev(uvnext(uv)) == uv && uvnext(uvprev(uv)) == uv,
117 "broken upvalue chain");
118 if (isgray(obj2gco(uv)))
119 gc_marktv(g, uvval(uv));
120 }
121}

123/* Mark userdata in mmudata list. */
124static void gc_mark_mmudata(global_State *g)
125{
126 GCobj *root = gcref(g->gc.mmudata);
127 GCobj *u = root;
128 if (u) {
129 do {
130 u = gcnext(u);
131 makewhite(g, u); /* Could be from previous GC. */
132 gc_mark(g, u);
133 } while (u != root);
134 }
135}

137/* Separate userdata objects to be finalized to mmudata list. */
138size_t lj_gc_separateudata(global_State *g, int all)
139{
140 size_t m = 0;
141 GCRef *p = &mainthread(g)->nextgc;
142 GCobj *o;
143 while ((o = gcref(*p)) != NULL) {
144 if (!(iswhite(o) || all) || isfinalized(gco2ud(o))) {
145 p = &o->gch.nextgc; /* Nothing to do. */
146 } else if (!lj_meta_fastg(g, tabref(gco2ud(o)->metatable), MM_gc)) {
147 markfinalized(o); /* Done, as there's no __gc metamethod. */
148 p = &o->gch.nextgc;
149 } else { /* Otherwise move userdata to be finalized to mmudata list. */
150 m += sizeudata(gco2ud(o));
151 markfinalized(o);
152 *p = o->gch.nextgc;
153 if (gcref(g->gc.mmudata)) { /* Link to end of mmudata list. */
154 GCobj *root = gcref(g->gc.mmudata);
155 setgcrefr(o->gch.nextgc, root->gch.nextgc);
156 setgcref(root->gch.nextgc, o);
157 setgcref(g->gc.mmudata, o);
158 } else { /* Create circular list. */
159 setgcref(o->gch.nextgc, o);
160 setgcref(g->gc.mmudata, o);
161 }
162 }
163 }
164 return m;
165}

169/* Traverse a table. */
170static int gc_traverse_tab(global_State *g, GCtab *t)
171{
172 int weak = 0;
173 cTValue *mode;
174 GCtab *mt = tabref(t->metatable);
175 if (mt)
176 gc_markobj(g, mt);
177 mode = lj_meta_fastg(g, mt, MM_mode);
178 if (mode && tvisstr(mode)) { /* Valid __mode field? */
179 const char *modestr = strVdata(mode);
180 int c;
181 while ((c = *modestr++)) {
182 if (c == 'k') weak |= LJ_GC_WEAKKEY;
183 else if (c == 'v') weak |= LJ_GC_WEAKVAL;
184 }
185 if (weak) { /* Weak tables are cleared in the atomic phase. */
186#if LJ_HASFFI
187 CTState *cts = ctype_ctsG(g);
188 if (cts && cts->finalizer == t) {
189 weak = (int)(~0u & ~LJ_GC_WEAKVAL);
190 } else
191#endif
192 {
193 t->marked = (uint8_t)((t->marked & ~LJ_GC_WEAK) | weak);
194 setgcrefr(t->gclist, g->gc.weak);
195 setgcref(g->gc.weak, obj2gco(t));
196 }
197 }
198 }
199 if (weak == LJ_GC_WEAK) /* Nothing to mark if both keys/values are weak. */
200 return 1;
201 if (!(weak & LJ_GC_WEAKVAL)) { /* Mark array part. */
202 MSize i, asize = t->asize;
203 for (i = 0; i < asize; i++)
204 gc_marktv(g, arrayslot(t, i));
205 }
206 if (t->hmask > 0) { /* Mark hash part. */
207 Node *node = noderef(t->node);
208 MSize i, hmask = t->hmask;
209 for (i = 0; i <= hmask; i++) {
210 Node *n = &node[i];
211 if (!tvisnil(&n->val)) { /* Mark non-empty slot. */
212 lj_assertG(!tvisnil(&n->key), "mark of nil key in non-empty slot");
213 if (!(weak & LJ_GC_WEAKKEY)) gc_marktv(g, &n->key);
214 if (!(weak & LJ_GC_WEAKVAL)) gc_marktv(g, &n->val);
215 }
216 }
217 }
218 return weak;
219}

221/* Traverse a function. */
222static void gc_traverse_func(global_State *g, GCfunc *fn)
223{
224 gc_markobj(g, tabref(fn->c.env));
225 if (isluafunc(fn)) {
226 uint32_t i;
227 lj_assertG(fn->l.nupvalues <= funcproto(fn)->sizeuv,
228 "function upvalues out of range");
229 gc_markobj(g, funcproto(fn));
230 for (i = 0; i < fn->l.nupvalues; i++) /* Mark Lua function upvalues. */
231 gc_markobj(g, &gcref(fn->l.uvptr[i])->uv);
232 } else {
233 uint32_t i;
234 for (i = 0; i < fn->c.nupvalues; i++) /* Mark C function upvalues. */
235 gc_marktv(g, &fn->c.upvalue[i]);
236 }
237}

252/* Traverse a trace. */
253static void gc_traverse_trace(global_State *g, GCtrace *T)
254{
255 IRRef ref;
256 if (T->traceno == 0) return;
257 for (ref = T->nk; ref < REF_TRUE; ref++) {
258 IRIns *ir = &T->ir[ref];
259 if (ir->o == IR_KGC)
260 gc_markobj(g, ir_kgc(ir));
261 if (irt_is64(ir->t) && ir->o != IR_KNULL)
262 ref++;
263 }
264 if (T->link) gc_marktrace(g, T->link);
265 if (T->nextroot) gc_marktrace(g, T->nextroot);
266 if (T->nextside) gc_marktrace(g, T->nextside);
267 gc_markobj(g, gcref(T->startpt));
268}

276/* Traverse a prototype. */
277static void gc_traverse_proto(global_State *g, GCproto *pt)
278{
279 ptrdiff_t i;
280 gc_mark_str(proto_chunkname(pt));
281 for (i = -(ptrdiff_t)pt->sizekgc; i < 0; i++) /* Mark collectable consts. */

288/* Traverse the frame structure of a stack. */
289static MSize gc_traverse_frames(global_State *g, lua_State *th)
290{
291 TValue *frame, *top = th->top-1, *bot = tvref(th->stack);
292 /* Note: extra vararg frame not skipped, marks function twice (harmless). */
293 for (frame = th->base-1; frame > bot+LJ_FR2; frame = frame_prev(frame)) {
294 GCfunc *fn = frame_func(frame);
295 TValue *ftop = frame;
296 if (isluafunc(fn)) ftop += funcproto(fn)->framesize;
297 if (ftop > top) top = ftop;
298 if (!LJ_FR2) gc_markobj(g, fn); /* Need to mark hidden function (or L). */
299 }
300 top++; /* Correct bias of -1 (frame == base-1). */
301 if (top > tvref(th->maxstack)) top = tvref(th->maxstack);
302 return (MSize)(top - bot); /* Return minimum needed stack size. */
303}
304
305/* Traverse a thread object. */
306static void gc_traverse_thread(global_State *g, lua_State *th)
307{
308 TValue *o, *top = th->top;
309 for (o = tvref(th->stack)+1+LJ_FR2; o < top; o++)
310 gc_marktv(g, o);
311 if (g->gc.state == GCSatomic) {
312 top = tvref(th->stack) + th->stacksize;
313 for (; o < top; o++) /* Clear unmarked slots. */
314 setnilV(o);
315 }
316 gc_markobj(g, tabref(th->env));
317 lj_state_shrinkstack(th, gc_traverse_frames(g, th));
318}

320/* Propagate one gray object. Traverse it and turn it black. */
321static size_t propagatemark(global_State *g)
322{
323 GCobj *o = gcref(g->gc.gray);
324 int gct = o->gch.gct;
325 lj_assertG(isgray(o), "propagation of non-gray object");
326 gray2black(o);
327 setgcrefr(g->gc.gray, o->gch.gclist); /* Remove from gray list. */
328 if (LJ_LIKELY(gct == ~LJ_TTAB)) {
329 GCtab *t = gco2tab(o);
330 if (gc_traverse_tab(g, t) > 0)
331 black2gray(o); /* Keep weak tables gray. */
332 return sizeof(GCtab) + sizeof(TValue) * t->asize +
333 (t->hmask ? sizeof(Node) * (t->hmask + 1) : 0);
334 } else if (LJ_LIKELY(gct == ~LJ_TFUNC)) {
335 GCfunc *fn = gco2func(o);
336 gc_traverse_func(g, fn);
337 return isluafunc(fn) ? sizeLfunc((MSize)fn->l.nupvalues) :
338 sizeCfunc((MSize)fn->c.nupvalues);
339 } else if (LJ_LIKELY(gct == ~LJ_TPROTO)) {
340 GCproto *pt = gco2pt(o);
341 gc_traverse_proto(g, pt);
342 return pt->sizept;
343 } else if (LJ_LIKELY(gct == ~LJ_TTHREAD)) {
344 lua_State *th = gco2th(o);
345 setgcrefr(th->gclist, g->gc.grayagain);
346 setgcref(g->gc.grayagain, o);
347 black2gray(o); /* Threads are never black. */
348 gc_traverse_thread(g, th);
349 return sizeof(lua_State) + sizeof(TValue) * th->stacksize;
350 } else {
351#if LJ_HASJIT
352 GCtrace *T = gco2trace(o);
353 gc_traverse_trace(g, T);
354 return ((sizeof(GCtrace)+7)&~7) + (T->nins-T->nk)*sizeof(IRIns) +
355 T->nsnap*sizeof(SnapShot) + T->nsnapmap*sizeof(SnapEntry);
356#else
357 lj_assertG(0, "bad GC type %d", gct);
358 return 0;
359#endif
360 }
361}

363/* Propagate all gray objects. */
364static size_t gc_propagate_gray(global_State *g)
365{
366 size_t m = 0;
367 while (gcref(g->gc.gray) != NULL)
368 m += propagatemark(g);
369 return m;

398/* Full sweep of a GC list. */
399#define gc_fullsweep(g, p) gc_sweep(g, (p), ~(uint32_t)0)
400
401/* Partial sweep of a GC list. */
402static GCRef *gc_sweep(global_State *g, GCRef *p, uint32_t lim)
403{
404 /* Mask with other white and LJ_GC_FIXED. Or LJ_GC_SFIXED on shutdown. */
405 int ow = otherwhite(g);
406 GCobj *o;
407 while ((o = gcref(*p)) != NULL && lim-- > 0) {
408 if (o->gch.gct == ~LJ_TTHREAD) /* Need to sweep open upvalues, too. */
409 gc_fullsweep(g, &gco2th(o)->openupval);
410 if (((o->gch.marked ^ LJ_GC_WHITES) & ow)) { /* Black or current white? */
411 lj_assertG(!isdead(g, o) || (o->gch.marked & LJ_GC_FIXED),
412 "sweep of undead object");
413 makewhite(g, o); /* Value is alive, change to the current white. */
414 p = &o->gch.nextgc;
415 } else { /* Otherwise value is dead, free it. */
416 lj_assertG(isdead(g, o) || ow == LJ_GC_SFIXED,
417 "sweep of unlive object");
418 setgcrefr(*p, o->gch.nextgc);
419 if (o == gcref(g->gc.root))
420 setgcrefr(g->gc.root, o->gch.nextgc); /* Adjust list anchor. */
421 gc_freefunc[o->gch.gct - ~LJ_TSTR](g, o);
422 }
423 }
424 return p;
425}

427/* Sweep one string interning table chain. Preserves hashalg bit. */
428static void gc_sweepstr(global_State *g, GCRef *chain)
429{
430 /* Mask with other white and LJ_GC_FIXED. Or LJ_GC_SFIXED on shutdown. */
431 int ow = otherwhite(g);
432 uintptr_t u = gcrefu(*chain);
433 GCRef q;
434 GCRef *p = &q;
435 GCobj *o;
436 setgcrefp(q, (u & ~(uintptr_t)1));
437 while ((o = gcref(*p)) != NULL) {
438 if (((o->gch.marked ^ LJ_GC_WHITES) & ow)) { /* Black or current white? */
439 lj_assertG(!isdead(g, o) || (o->gch.marked & LJ_GC_FIXED),
440 "sweep of undead string");
441 makewhite(g, o); /* String is alive, change to the current white. */
442 p = &o->gch.nextgc;
443 } else { /* Otherwise string is dead, free it. */
444 lj_assertG(isdead(g, o) || ow == LJ_GC_SFIXED,
445 "sweep of unlive string");
446 setgcrefr(*p, o->gch.nextgc);
447 lj_str_free(g, gco2str(o));
448 }
449 }
450 setgcrefp(*chain, (gcrefu(q) | (u & 1)));
451}

453/* Check whether we can clear a key or a value slot from a table. */
454static int gc_mayclear(cTValue *o, int val)
455{
456 if (tvisgcv(o)) { /* Only collectable objects can be weak references. */
457 if (tvisstr(o)) { /* But strings cannot be used as weak references. */
458 gc_mark_str(strV(o)); /* And need to be marked. */
459 return 0;
460 }
461 if (iswhite(gcV(o)))
462 return 1; /* Object is about to be collected. */
463 if (tvisudata(o) && val && isfinalized(udataV(o)))
464 return 1; /* Finalized userdata is dropped only from values. */
465 }
466 return 0; /* Cannot clear. */
467}

469/* Clear collected entries from weak tables. */
470static void gc_clearweak(global_State *g, GCobj *o)
471{
472 UNUSED(g);
473 while (o) {
474 GCtab *t = gco2tab(o);
475 lj_assertG((t->marked & LJ_GC_WEAK), "clear of non-weak table");
476 if ((t->marked & LJ_GC_WEAKVAL)) {
477 MSize i, asize = t->asize;
478 for (i = 0; i < asize; i++) {
479 /* Clear array slot when value is about to be collected. */
480 TValue *tv = arrayslot(t, i);
481 if (gc_mayclear(tv, 1))
482 setnilV(tv);
483 }
484 }
485 if (t->hmask > 0) {
486 Node *node = noderef(t->node);
487 MSize i, hmask = t->hmask;
488 for (i = 0; i <= hmask; i++) {
489 Node *n = &node[i];
490 /* Clear hash slot when key or value is about to be collected. */
491 if (!tvisnil(&n->val) && (gc_mayclear(&n->key, 0) ||
492 gc_mayclear(&n->val, 1)))
493 setnilV(&n->val);
494 }
495 }
496 o = gcref(t->gclist);
497 }
498}

500/* Call a userdata or cdata finalizer. */
501static void gc_call_finalizer(global_State *g, lua_State *L,
502 cTValue *mo, GCobj *o)
503{
504 /* Save and restore lots of state around the __gc callback. */
505 uint8_t oldh = hook_save(g);
506 GCSize oldt = g->gc.threshold;
507 int errcode;
508 TValue *top;
509 lj_trace_abort(g);
510 hook_entergc(g); /* Disable hooks and new traces during __gc. */
511 if (LJ_HASPROFILE && (oldh & HOOK_PROFILE)) lj_dispatch_update(g);
512 g->gc.threshold = LJ_MAX_MEM; /* Prevent GC steps. */
513 top = L->top;
514 copyTV(L, top++, mo);
515 if (LJ_FR2) setnilV(top++);
516 setgcV(L, top, o, ~o->gch.gct);
517 L->top = top+1;
518 errcode = lj_vm_pcall(L, top, 1+0, -1); /* Stack: |mo|o| -> | */
519 hook_restore(g, oldh);
520 if (LJ_HASPROFILE && (oldh & HOOK_PROFILE)) lj_dispatch_update(g);
521 g->gc.threshold = oldt; /* Restore GC threshold. */
522 if (errcode)
523 lj_err_throw(L, errcode); /* Propagate errors. */
524}

526/* Finalize one userdata or cdata object from the mmudata list. */
527static void gc_finalize(lua_State *L)
528{
529 global_State *g = G(L);
530 GCobj *o = gcnext(gcref(g->gc.mmudata));
531 cTValue *mo;
532 lj_assertG(tvref(g->jit_base) == NULL, "finalizer called on trace");
533 /* Unchain from list of userdata to be finalized. */
534 if (o == gcref(g->gc.mmudata))
535 setgcrefnull(g->gc.mmudata);
536 else
537 setgcrefr(gcref(g->gc.mmudata)->gch.nextgc, o->gch.nextgc);
538#if LJ_HASFFI
539 if (o->gch.gct == ~LJ_TCDATA) {
540 TValue tmp, *tv;
541 /* Add cdata back to the GC list and make it white. */
542 setgcrefr(o->gch.nextgc, g->gc.root);
543 setgcref(g->gc.root, o);
544 makewhite(g, o);
545 o->gch.marked &= (uint8_t)~LJ_GC_CDATA_FIN;
546 /* Resolve finalizer. */
547 setcdataV(L, &tmp, gco2cd(o));
548 tv = lj_tab_set(L, ctype_ctsG(g)->finalizer, &tmp);
549 if (!tvisnil(tv)) {
550 g->gc.nocdatafin = 0;
551 copyTV(L, &tmp, tv);
552 setnilV(tv); /* Clear entry in finalizer table. */
553 gc_call_finalizer(g, L, &tmp, o);
554 }
555 return;
556 }
557#endif
558 /* Add userdata back to the main userdata list and make it white. */
559 setgcrefr(o->gch.nextgc, mainthread(g)->nextgc);
560 setgcref(mainthread(g)->nextgc, o);
561 makewhite(g, o);
562 /* Resolve the __gc metamethod. */
563 mo = lj_meta_fastg(g, tabref(gco2ud(o)->metatable), MM_gc);
564 if (mo)
565 gc_call_finalizer(g, L, mo, o);
566}

600/* Free all remaining GC objects. */
601void lj_gc_freeall(global_State *g)
602{
603 MSize i, strmask;
604 /* Free everything, except super-fixed objects (the main thread). */
605 g->gc.currentwhite = LJ_GC_WHITES | LJ_GC_SFIXED;
606 gc_fullsweep(g, &g->gc.root);
607 strmask = g->str.mask;
608 for (i = 0; i <= strmask; i++) /* Free all string hash chains. */
609 gc_sweepstr(g, &g->str.tab[i]);
610}

614/* Atomic part of the GC cycle, transitioning from mark to sweep phase. */
615static void atomic(global_State *g, lua_State *L)
616{
617 size_t udsize;
618
619 gc_mark_uv(g); /* Need to remark open upvalues (the thread may be dead). */
620 gc_propagate_gray(g); /* Propagate any left-overs. */
621
622 setgcrefr(g->gc.gray, g->gc.weak); /* Empty the list of weak tables. */
623 setgcrefnull(g->gc.weak);
624 lj_assertG(!iswhite(obj2gco(mainthread(g))), "main thread turned white");
625 gc_markobj(g, L); /* Mark running thread. */
626 gc_traverse_curtrace(g); /* Traverse current trace. */
627 gc_mark_gcroot(g); /* Mark GC roots (again). */
628 gc_propagate_gray(g); /* Propagate all of the above. */
629
630 setgcrefr(g->gc.gray, g->gc.grayagain); /* Empty the 2nd chance list. */
631 setgcrefnull(g->gc.grayagain);
632 gc_propagate_gray(g); /* Propagate it. */
633
634 udsize = lj_gc_separateudata(g, 0); /* Separate userdata to be finalized. */
635 gc_mark_mmudata(g); /* Mark them. */
636 udsize += gc_propagate_gray(g); /* And propagate the marks. */
637
638 /* All marking done, clear weak tables. */
639 gc_clearweak(g, gcref(g->gc.weak));
640
641 lj_buf_shrink(L, &g->tmpbuf); /* Shrink temp buffer. */
642
643 /* Prepare for sweep phase. */
644 g->gc.currentwhite = (uint8_t)otherwhite(g); /* Flip current white. */
645 g->strempty.marked = g->gc.currentwhite;
646 setmref(g->gc.sweep, &g->gc.root);
647 g->gc.estimate = g->gc.total - (GCSize)udsize; /* Initial estimate. */
648}
649
650/* GC state machine. Returns a cost estimate for each step performed. */
651static size_t gc_onestep(lua_State *L)
652{
653 global_State *g = G(L);
654 switch (g->gc.state) {
655 case GCSpause:
656 gc_mark_start(g); /* Start a new GC cycle by marking all GC roots. */
657 return 0;
658 case GCSpropagate:
659 if (gcref(g->gc.gray) != NULL)
660 return propagatemark(g); /* Propagate one gray object. */
661 g->gc.state = GCSatomic; /* End of mark phase. */
662 return 0;
663 case GCSatomic:
664 if (tvref(g->jit_base)) /* Don't run atomic phase on trace. */
665 return LJ_MAX_MEM;
666 atomic(g, L);
667 g->gc.state = GCSsweepstring; /* Start of sweep phase. */
668 g->gc.sweepstr = 0;
669 return 0;
670 case GCSsweepstring: {
671 GCSize old = g->gc.total;
672 gc_sweepstr(g, &g->str.tab[g->gc.sweepstr++]); /* Sweep one chain. */
673 if (g->gc.sweepstr > g->str.mask)
674 g->gc.state = GCSsweep; /* All string hash chains sweeped. */
675 lj_assertG(old >= g->gc.total, "sweep increased memory");
676 g->gc.estimate -= old - g->gc.total;
677 return GCSWEEPCOST;
678 }
679 case GCSsweep: {
680 GCSize old = g->gc.total;
681 setmref(g->gc.sweep, gc_sweep(g, mref(g->gc.sweep, GCRef), GCSWEEPMAX));
682 lj_assertG(old >= g->gc.total, "sweep increased memory");
683 g->gc.estimate -= old - g->gc.total;
684 if (gcref(*mref(g->gc.sweep, GCRef)) == NULL) {
685 if (g->str.num <= (g->str.mask >> 2) && g->str.mask > LJ_MIN_STRTAB*2-1)
686 lj_str_resize(L, g->str.mask >> 1); /* Shrink string table. */
687 if (gcref(g->gc.mmudata)) { /* Need any finalizations? */
688 g->gc.state = GCSfinalize;
689#if LJ_HASFFI
690 g->gc.nocdatafin = 1;
691#endif
692 } else { /* Otherwise skip this phase to help the JIT. */
693 g->gc.state = GCSpause; /* End of GC cycle. */
694 g->gc.debt = 0;
695 }
696 }
697 return GCSWEEPMAX*GCSWEEPCOST;
698 }
699 case GCSfinalize:
700 if (gcref(g->gc.mmudata) != NULL) {
701 if (tvref(g->jit_base)) /* Don't call finalizers on trace. */
702 return LJ_MAX_MEM;
703 gc_finalize(L); /* Finalize one userdata object. */
704 if (g->gc.estimate > GCFINALIZECOST)
705 g->gc.estimate -= GCFINALIZECOST;
706 return GCFINALIZECOST;
707 }
708#if LJ_HASFFI
709 if (!g->gc.nocdatafin) lj_tab_rehash(L, ctype_ctsG(g)->finalizer);
710#endif
711 g->gc.state = GCSpause; /* End of GC cycle. */
712 g->gc.debt = 0;
713 return 0;
714 default:
715 lj_assertG(0, "bad GC state");
716 return 0;
717 }
718}

720/* Perform a limited amount of incremental GC steps. */
721int LJ_FASTCALL lj_gc_step(lua_State *L)
722{
723 global_State *g = G(L);
724 GCSize lim;
725 int32_t ostate = g->vmstate;
726 setvmstate(g, GC);
727 lim = (GCSTEPSIZE/100) * g->gc.stepmul;
728 if (lim == 0)
729 lim = LJ_MAX_MEM;
730 if (g->gc.total > g->gc.threshold)
731 g->gc.debt += g->gc.total - g->gc.threshold;
732 do {
733 lim -= (GCSize)gc_onestep(L);
734 if (g->gc.state == GCSpause) {
735 g->gc.threshold = (g->gc.estimate/100) * g->gc.pause;
736 g->vmstate = ostate;
737 return 1; /* Finished a GC cycle. */
738 }
739 } while (sizeof(lim) == 8 ? ((int64_t)lim > 0) : ((int32_t)lim > 0));
740 if (g->gc.debt < GCSTEPSIZE) {
741 g->gc.threshold = g->gc.total + GCSTEPSIZE;
742 g->vmstate = ostate;
743 return -1;
744 } else {
745 g->gc.debt -= GCSTEPSIZE;
746 g->gc.threshold = g->gc.total;
747 g->vmstate = ostate;
748 return 0;
749 }
750}

753void LJ_FASTCALL lj_gc_step_fixtop(lua_State *L)
754{
755 if (curr_funcisL(L)) L->top = curr_topL(L);
756 lj_gc_step(L);

761int LJ_FASTCALL lj_gc_step_jit(global_State *g, MSize steps)

773/* Perform a full GC cycle. */
774void lj_gc_fullgc(lua_State *L)
775{
776 global_State *g = G(L);
777 int32_t ostate = g->vmstate;
778 setvmstate(g, GC);
779 if (g->gc.state <= GCSatomic) { /* Caught somewhere in the middle. */
780 setmref(g->gc.sweep, &g->gc.root); /* Sweep everything (preserving it). */
781 setgcrefnull(g->gc.gray); /* Reset lists from partial propagation. */
782 setgcrefnull(g->gc.grayagain);
783 setgcrefnull(g->gc.weak);
784 g->gc.state = GCSsweepstring; /* Fast forward to the sweep phase. */
785 g->gc.sweepstr = 0;
786 }
787 while (g->gc.state == GCSsweepstring || g->gc.state == GCSsweep)
788 gc_onestep(L); /* Finish sweep. */
789 lj_assertG(g->gc.state == GCSfinalize || g->gc.state == GCSpause,
790 "bad GC state");
791 /* Now perform a full GC. */
792 g->gc.state = GCSpause;
793 do { gc_onestep(L); } while (g->gc.state != GCSpause);
794 g->gc.threshold = (g->gc.estimate/100) * g->gc.pause;
795 g->vmstate = ostate;
796}

798/* -- Write barriers ------------------------------------------------------ */
799
800/* Move the GC propagation frontier forward. */
801void lj_gc_barrierf(global_State *g, GCobj *o, GCobj *v)
802{
803 lj_assertG(isblack(o) && iswhite(v) && !isdead(g, v) && !isdead(g, o),
804 "bad object states for forward barrier");
805 lj_assertG(g->gc.state != GCSfinalize && g->gc.state != GCSpause,
806 "bad GC state");
807 lj_assertG(o->gch.gct != ~LJ_TTAB, "barrier object is not a table");
808 /* Preserve invariant during propagation. Otherwise it doesn't matter. */
809 if (g->gc.state == GCSpropagate || g->gc.state == GCSatomic)
810 gc_mark(g, v); /* Move frontier forward. */
811 else
812 makewhite(g, o); /* Make it white to avoid the following barrier. */
813}
814
815/* Specialized barrier for closed upvalue. Pass &uv->tv. */
816void LJ_FASTCALL lj_gc_barrieruv(global_State *g, TValue *tv)
817{
818#define TV2MARKED(x) \
819 (*((uint8_t *)(x) - offsetof(GCupval, tv) + offsetof(GCupval, marked)))
820 if (g->gc.state == GCSpropagate || g->gc.state == GCSatomic)
821 gc_mark(g, gcV(tv));
822 else
823 TV2MARKED(tv) = (TV2MARKED(tv) & (uint8_t)~LJ_GC_COLORS) | curwhite(g);
824#undef TV2MARKED
825}
826
827/* Close upvalue. Also needs a write barrier. */
828void lj_gc_closeuv(global_State *g, GCupval *uv)
829{
830 GCobj *o = obj2gco(uv);
831 /* Copy stack slot to upvalue itself and point to the copy. */
832 copyTV(mainthread(g), &uv->tv, uvval(uv));
833 setmref(uv->v, &uv->tv);
834 uv->closed = 1;
835 setgcrefr(o->gch.nextgc, g->gc.root);
836 setgcref(g->gc.root, o);
837 if (isgray(o)) { /* A closed upvalue is never gray, so fix this. */
838 if (g->gc.state == GCSpropagate || g->gc.state == GCSatomic) {
839 gray2black(o); /* Make it black and preserve invariant. */
840 if (tviswhite(&uv->tv))
841 lj_gc_barrierf(g, o, gcV(&uv->tv));
842 } else {
843 makewhite(g, o); /* Make it white, i.e. sweep the upvalue. */
844 lj_assertG(g->gc.state != GCSfinalize && g->gc.state != GCSpause,
845 "bad GC state");
846 }
847 }
848}

861/* Call pluggable memory allocator to allocate or resize a fragment. */
862void *lj_mem_realloc(lua_State *L, void *p, GCSize osz, GCSize nsz)
863{
864 global_State *g = G(L);
865 lj_assertG((osz == 0) == (p == NULL), "realloc API violation");
866 p = g->allocf(g->allocd, p, osz, nsz);
867 if (p == NULL && nsz > 0)
868 lj_err_mem(L);
869 lj_assertG((nsz == 0) == (p == NULL), "allocf API violation");
870 lj_assertG(checkptrGC(p),
871 "allocated memory address %p outside required range", p);
872 g->gc.total = (g->gc.total - osz) + nsz;
873 return p;
874}

876/* Allocate new GC object and link it to the root set. */
877void * LJ_FASTCALL lj_mem_newgco(lua_State *L, GCSize size)
878{
879 global_State *g = G(L);
880 GCobj *o = (GCobj *)g->allocf(g->allocd, NULL, 0, size);
881 if (o == NULL)
882 lj_err_mem(L);
883 lj_assertG(checkptrGC(o),
884 "allocated memory address %p outside required range", o);
885 g->gc.total += size;
886 setgcrefr(o->gch.nextgc, g->gc.root);
887 setgcref(g->gc.root, o);
888 newwhite(g, o);
889 return o;
890}

892/* Resize growable vector. */
893void *lj_mem_grow(lua_State *L, void *p, MSize *szp, MSize lim, MSize esz)
894{
895 MSize sz = (*szp) << 1;
896 if (sz < LJ_MIN_VECSZ)
897 sz = LJ_MIN_VECSZ;
898 if (sz > lim)
899 sz = lim;
900 p = lj_mem_realloc(L, p, (*szp)*esz, sz*esz);
901 *szp = sz;
902 return p;
903}
