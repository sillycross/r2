29/* Pass IR on to next optimization in chain (FOLD). */
30#define emitir(ot, a, b) (lj_ir_set(J, (ot), (a), (b)), lj_opt_fold(J))

32/* Emit raw IR without passing through optimizations. */
33#define emitir_raw(ot, a, b) (lj_ir_set(J, (ot), (a), (b)), lj_ir_emit(J))

37/* Grow snapshot buffer. */
38void lj_snap_grow_buf_(jit_State *J, MSize need)
39{
40 MSize maxsnap = (MSize)J->param[JIT_P_maxsnap];
41 if (need > maxsnap)
42 lj_trace_err(J, LJ_TRERR_SNAPOV);
43 lj_mem_growvec(J->L, J->snapbuf, J->sizesnap, maxsnap, SnapShot);
44 J->cur.snap = J->snapbuf;
45}
46
47/* Grow snapshot map buffer. */
48void lj_snap_grow_map_(jit_State *J, MSize need)
49{
50 if (need < 2*J->sizesnapmap)
51 need = 2*J->sizesnapmap;
52 else if (need < 64)
53 need = 64;
54 J->snapmapbuf = (SnapEntry *)lj_mem_realloc(J->L, J->snapmapbuf,
55 J->sizesnapmap*sizeof(SnapEntry), need*sizeof(SnapEntry));
56 J->cur.snapmap = J->snapmapbuf;
57 J->sizesnapmap = need;
58}

62/* Add all modified slots to the snapshot. */
63static MSize snapshot_slots(jit_State *J, SnapEntry *map, BCReg nslots)
64{
65 IRRef retf = J->chain[IR_RETF]; /* Limits SLOAD restore elimination. */
66 BCReg s;
67 MSize n = 0;
68 for (s = 0; s < nslots; s++) {
69 TRef tr = J->slot[s];
70 IRRef ref = tref_ref(tr);
71#if LJ_FR2
72 if (s == 1) { /* Ignore slot 1 in LJ_FR2 mode, except if tailcalled. */
73 if ((tr & TREF_FRAME))
74 map[n++] = SNAP(1, SNAP_FRAME | SNAP_NORESTORE, REF_NIL);
75 continue;
76 }
77 if ((tr & (TREF_FRAME | TREF_CONT)) && !ref) {
78 cTValue *base = J->L->base - J->baseslot;
79 tr = J->slot[s] = (tr & 0xff0000) | lj_ir_k64(J, IR_KNUM, base[s].u64);
80 ref = tref_ref(tr);
81 }
82#endif
83 if (ref) {
84 SnapEntry sn = SNAP_TR(s, tr);
85 IRIns *ir = &J->cur.ir[ref];
86 if ((LJ_FR2 || !(sn & (SNAP_CONT|SNAP_FRAME))) &&
87 ir->o == IR_SLOAD && ir->op1 == s && ref > retf) {
88 /*
89 ** No need to snapshot unmodified non-inherited slots.
90 ** But always snapshot the function below a frame in LJ_FR2 mode.
91 */
92 if (!(ir->op2 & IRSLOAD_INHERIT) &&
93 (!LJ_FR2 || s == 0 || s+1 == nslots ||
94 !(J->slot[s+1] & (TREF_CONT|TREF_FRAME))))
95 continue;
96 /* No need to restore readonly slots and unmodified non-parent slots. */
97 if (!(LJ_DUALNUM && (ir->op2 & IRSLOAD_CONVERT)) &&
98 (ir->op2 & (IRSLOAD_READONLY|IRSLOAD_PARENT)) != IRSLOAD_PARENT)
99 sn |= SNAP_NORESTORE;
100 }
101 if (LJ_SOFTFP32 && irt_isnum(ir->t))
102 sn |= SNAP_SOFTFPNUM;
103 map[n++] = sn;
104 }
105 }
106 return n;
107}

109/* Add frame links at the end of the snapshot. */
110static MSize snapshot_framelinks(jit_State *J, SnapEntry *map, uint8_t *topslot)
111{
112 cTValue *frame = J->L->base - 1;
113 cTValue *lim = J->L->base - J->baseslot + LJ_FR2;
114 GCfunc *fn = frame_func(frame);
115 cTValue *ftop = isluafunc(fn) ? (frame+funcproto(fn)->framesize) : J->L->top;
116#if LJ_FR2
117 uint64_t pcbase = (u64ptr(J->pc) << 8) | (J->baseslot - 2);
118 lj_assertJ(2 <= J->baseslot && J->baseslot <= 257, "bad baseslot");
119 memcpy(map, &pcbase, sizeof(uint64_t));
120#else
121 MSize f = 0;
122 map[f++] = SNAP_MKPC(J->pc); /* The current PC is always the first entry. */
123#endif
124 lj_assertJ(!J->pt ||
125 (J->pc >= proto_bc(J->pt) &&
126 J->pc < proto_bc(J->pt) + J->pt->sizebc), "bad snapshot PC");
127 while (frame > lim) { /* Backwards traversal of all frames above base. */
128 if (frame_islua(frame)) {
129#if !LJ_FR2
130 map[f++] = SNAP_MKPC(frame_pc(frame));
131#endif
132 frame = frame_prevl(frame);
133 } else if (frame_iscont(frame)) {
134#if !LJ_FR2
135 map[f++] = SNAP_MKFTSZ(frame_ftsz(frame));
136 map[f++] = SNAP_MKPC(frame_contpc(frame));
137#endif
138 frame = frame_prevd(frame);
139 } else {
140 lj_assertJ(!frame_isc(frame), "broken frame chain");
141#if !LJ_FR2
142 map[f++] = SNAP_MKFTSZ(frame_ftsz(frame));
143#endif
144 frame = frame_prevd(frame);
145 continue;
146 }
147 if (frame + funcproto(frame_func(frame))->framesize > ftop)
148 ftop = frame + funcproto(frame_func(frame))->framesize;
149 }
150 *topslot = (uint8_t)(ftop - lim);
151#if LJ_FR2
152 lj_assertJ(sizeof(SnapEntry) * 2 == sizeof(uint64_t), "bad SnapEntry def");
153 return 2;
154#else
155 lj_assertJ(f == (MSize)(1 + J->framedepth), "miscalculated snapshot size");
156 return f;
157#endif
158}

160/* Take a snapshot of the current stack. */
161static void snapshot_stack(jit_State *J, SnapShot *snap, MSize nsnapmap)
162{
163 BCReg nslots = J->baseslot + J->maxslot;
164 MSize nent;
165 SnapEntry *p;
166 /* Conservative estimate. */
167 lj_snap_grow_map(J, nsnapmap + nslots + (MSize)(LJ_FR2?2:J->framedepth+1));
168 p = &J->cur.snapmap[nsnapmap];
169 nent = snapshot_slots(J, p, nslots);
170 snap->nent = (uint8_t)nent;
171 nent += snapshot_framelinks(J, p + nent, &snap->topslot);
172 snap->mapofs = (uint32_t)nsnapmap;
173 snap->ref = (IRRef1)J->cur.nins;
174 snap->mcofs = 0;
175 snap->nslots = (uint8_t)nslots;
176 snap->count = 0;
177 J->cur.nsnapmap = (uint32_t)(nsnapmap + nent);
178}

180/* Add or merge a snapshot. */
181void lj_snap_add(jit_State *J)
182{
183 MSize nsnap = J->cur.nsnap;
184 MSize nsnapmap = J->cur.nsnapmap;
185 /* Merge if no ins. inbetween or if requested and no guard inbetween. */
186 if ((nsnap > 0 && J->cur.snap[nsnap-1].ref == J->cur.nins) ||
187 (J->mergesnap && !irt_isguard(J->guardemit))) {
188 if (nsnap == 1) { /* But preserve snap #0 PC. */
189 emitir_raw(IRT(IR_NOP, IRT_NIL), 0, 0);
190 goto nomerge;
191 }
192 nsnapmap = J->cur.snap[--nsnap].mapofs;
193 } else {
194 nomerge:
195 lj_snap_grow_buf(J, nsnap+1);
196 J->cur.nsnap = (uint16_t)(nsnap+1);
197 }
198 J->mergesnap = 0;
199 J->guardemit.irt = 0;
200 snapshot_stack(J, &J->cur.snap[nsnap], nsnapmap);
201}

205#define SNAP_USEDEF_SLOTS (LJ_MAX_JSLOTS+LJ_STACK_EXTRA)
206
207/* Find unused slots with reaching-definitions bytecode data-flow analysis. */
208static BCReg snap_usedef(jit_State *J, uint8_t *udf,
209 const BCIns *pc, BCReg maxslot)
210{
211 BCReg s;
212 GCobj *o;
213
214 if (maxslot == 0) return 0;
215#ifdef LUAJIT_USE_VALGRIND
216 /* Avoid errors for harmless reads beyond maxslot. */
217 memset(udf, 1, SNAP_USEDEF_SLOTS);
218#else
219 memset(udf, 1, maxslot);
220#endif
221
222 /* Treat open upvalues as used. */
223 o = gcref(J->L->openupval);
224 while (o) {
225 if (uvval(gco2uv(o)) < J->L->base) break;
226 udf[uvval(gco2uv(o)) - J->L->base] = 0;
227 o = gcref(o->gch.nextgc);
228 }
229
230#define USE_SLOT(s) udf[(s)] &= ~1
231#define DEF_SLOT(s) udf[(s)] *= 3
232
233 /* Scan through following bytecode and check for uses/defs. */
234 lj_assertJ(pc >= proto_bc(J->pt) && pc < proto_bc(J->pt) + J->pt->sizebc,
235 "snapshot PC out of range");
236 for (;;) {
237 BCIns ins = *pc++;
238 BCOp op = bc_op(ins);
239 switch (bcmode_b(op)) {
240 case BCMvar: USE_SLOT(bc_b(ins)); break;
241 default: break;
242 }
243 switch (bcmode_c(op)) {
244 case BCMvar: USE_SLOT(bc_c(ins)); break;
245 case BCMrbase:
246 lj_assertJ(op == BC_CAT, "unhandled op %d with RC rbase", op);
247 for (s = bc_b(ins); s <= bc_c(ins); s++) USE_SLOT(s);
248 for (; s < maxslot; s++) DEF_SLOT(s);
249 break;
250 case BCMjump:
251 handle_jump: {
252 BCReg minslot = bc_a(ins);
253 if (op >= BC_FORI && op <= BC_JFORL) minslot += FORL_EXT;
254 else if (op >= BC_ITERL && op <= BC_JITERL) minslot += bc_b(pc[-2])-1;
255 else if (op == BC_UCLO) { pc += bc_j(ins); break; }
256 for (s = minslot; s < maxslot; s++) DEF_SLOT(s);
257 return minslot < maxslot ? minslot : maxslot;
258 }
259 case BCMlit:
260 if (op == BC_JFORL || op == BC_JITERL || op == BC_JLOOP) {
261 goto handle_jump;
262 } else if (bc_isret(op)) {
263 BCReg top = op == BC_RETM ? maxslot : (bc_a(ins) + bc_d(ins)-1);
264 for (s = 0; s < bc_a(ins); s++) DEF_SLOT(s);
265 for (; s < top; s++) USE_SLOT(s);
266 for (; s < maxslot; s++) DEF_SLOT(s);
267 return 0;
268 }
269 break;
270 case BCMfunc: return maxslot; /* NYI: will abort, anyway. */
271 default: break;
272 }
273 switch (bcmode_a(op)) {
274 case BCMvar: USE_SLOT(bc_a(ins)); break;
275 case BCMdst:
276 if (!(op == BC_ISTC || op == BC_ISFC)) DEF_SLOT(bc_a(ins));
277 break;
278 case BCMbase:
279 if (op >= BC_CALLM && op <= BC_ITERN) {
280 BCReg top = (op == BC_CALLM || op == BC_CALLMT || bc_c(ins) == 0) ?
281 maxslot : (bc_a(ins) + bc_c(ins)+LJ_FR2);
282 if (LJ_FR2) DEF_SLOT(bc_a(ins)+1);
283 s = bc_a(ins) - ((op == BC_ITERC || op == BC_ITERN) ? 3 : 0);
284 for (; s < top; s++) USE_SLOT(s);
285 for (; s < maxslot; s++) DEF_SLOT(s);
286 if (op == BC_CALLT || op == BC_CALLMT) {
287 for (s = 0; s < bc_a(ins); s++) DEF_SLOT(s);
288 return 0;
289 }
290 } else if (op == BC_VARG) {
291 return maxslot; /* NYI: punt. */
292 } else if (op == BC_KNIL) {
293 for (s = bc_a(ins); s <= bc_d(ins); s++) DEF_SLOT(s);
294 } else if (op == BC_TSETM) {
295 for (s = bc_a(ins)-1; s < maxslot; s++) USE_SLOT(s);
296 }
297 break;
298 default: break;
299 }
300 lj_assertJ(pc >= proto_bc(J->pt) && pc < proto_bc(J->pt) + J->pt->sizebc,
301 "use/def analysis PC out of range");
302 }
303
304#undef USE_SLOT
305#undef DEF_SLOT
306
307 return 0; /* unreachable */
308}

310/* Purge dead slots before the next snapshot. */
311void lj_snap_purge(jit_State *J)
312{
313 uint8_t udf[SNAP_USEDEF_SLOTS];
314 BCReg s, maxslot = J->maxslot;
315 if (bc_op(*J->pc) == BC_FUNCV && maxslot > J->pt->numparams)
316 maxslot = J->pt->numparams;
317 s = snap_usedef(J, udf, J->pc, maxslot);
318 for (; s < maxslot; s++)
319 if (udf[s] != 0)
320 J->base[s] = 0; /* Purge dead slots. */
321}

323/* Shrink last snapshot. */
324void lj_snap_shrink(jit_State *J)
325{
326 SnapShot *snap = &J->cur.snap[J->cur.nsnap-1];
327 SnapEntry *map = &J->cur.snapmap[snap->mapofs];
328 MSize n, m, nlim, nent = snap->nent;
329 uint8_t udf[SNAP_USEDEF_SLOTS];
330 BCReg maxslot = J->maxslot;
331 BCReg baseslot = J->baseslot;
332 BCReg minslot = snap_usedef(J, udf, snap_pc(&map[nent]), maxslot);
333 maxslot += baseslot;
334 minslot += baseslot;
335 snap->nslots = (uint8_t)maxslot;
336 for (n = m = 0; n < nent; n++) { /* Remove unused slots from snapshot. */
337 BCReg s = snap_slot(map[n]);
338 if (s < minslot || (s < maxslot && udf[s-baseslot] == 0))
339 map[m++] = map[n]; /* Only copy used slots. */
340 }
341 snap->nent = (uint8_t)m;
342 nlim = J->cur.nsnapmap - snap->mapofs - 1;
343 while (n <= nlim) map[m++] = map[n++]; /* Move PC + frame links down. */
344 J->cur.nsnapmap = (uint32_t)(snap->mapofs + m); /* Free up space in map. */
345}

349/* Initialize a Bloom Filter with all renamed refs.
350** There are very few renames (often none), so the filter has
351** very few bits set. This makes it suitable for negative filtering.
352*/
353static BloomFilter snap_renamefilter(GCtrace *T, SnapNo lim)
354{
355 BloomFilter rfilt = 0;
356 IRIns *ir;
357 for (ir = &T->ir[T->nins-1]; ir->o == IR_RENAME; ir--)
358 if (ir->op2 <= lim)
359 bloomset(rfilt, ir->op1);
360 return rfilt;
361}
362
363/* Process matching renames to find the original RegSP. */
364static RegSP snap_renameref(GCtrace *T, SnapNo lim, IRRef ref, RegSP rs)
365{
366 IRIns *ir;
367 for (ir = &T->ir[T->nins-1]; ir->o == IR_RENAME; ir--)
368 if (ir->op1 == ref && ir->op2 <= lim)
369 rs = ir->prev;
370 return rs;
371}
372
373/* Copy RegSP from parent snapshot to the parent links of the IR. */
374IRIns *lj_snap_regspmap(jit_State *J, GCtrace *T, SnapNo snapno, IRIns *ir)
375{
376 SnapShot *snap = &T->snap[snapno];
377 SnapEntry *map = &T->snapmap[snap->mapofs];
378 BloomFilter rfilt = snap_renamefilter(T, snapno);
379 MSize n = 0;
380 IRRef ref = 0;
381 UNUSED(J);
382 for ( ; ; ir++) {
383 uint32_t rs;
384 if (ir->o == IR_SLOAD) {
385 if (!(ir->op2 & IRSLOAD_PARENT)) break;
386 for ( ; ; n++) {
387 lj_assertJ(n < snap->nent, "slot %d not found in snapshot", ir->op1);
388 if (snap_slot(map[n]) == ir->op1) {
389 ref = snap_ref(map[n++]);
390 break;
391 }
392 }
393 } else if (LJ_SOFTFP32 && ir->o == IR_HIOP) {
394 ref++;
395 } else if (ir->o == IR_PVAL) {
396 ref = ir->op1 + REF_BIAS;
397 } else {
398 break;
399 }
400 rs = T->ir[ref].prev;
401 if (bloomtest(rfilt, ref))
402 rs = snap_renameref(T, snapno, ref, rs);
403 ir->prev = (uint16_t)rs;
404 lj_assertJ(regsp_used(rs), "unused IR %04d in snapshot", ref - REF_BIAS);
405 }
406 return ir;
407}

411/* Replay constant from parent trace. */
412static TRef snap_replay_const(jit_State *J, IRIns *ir)
413{
414 /* Only have to deal with constants that can occur in stack slots. */
415 switch ((IROp)ir->o) {
416 case IR_KPRI: return TREF_PRI(irt_type(ir->t));
417 case IR_KINT: return lj_ir_kint(J, ir->i);
418 case IR_KGC: return lj_ir_kgc(J, ir_kgc(ir), irt_t(ir->t));
419 case IR_KNUM: case IR_KINT64:
420 return lj_ir_k64(J, (IROp)ir->o, ir_k64(ir)->u64);
421 case IR_KPTR: return lj_ir_kptr(J, ir_kptr(ir)); /* Continuation. */
422 default: lj_assertJ(0, "bad IR constant op %d", ir->o); return TREF_NIL;
423 }
424}

426/* De-duplicate parent reference. */
427static TRef snap_dedup(jit_State *J, SnapEntry *map, MSize nmax, IRRef ref)
428{
429 MSize j;
430 for (j = 0; j < nmax; j++)
431 if (snap_ref(map[j]) == ref)
432 return J->slot[snap_slot(map[j])] & ~(SNAP_CONT|SNAP_FRAME);
433 return 0;
434}
435
436/* Emit parent reference with de-duplication. */
437static TRef snap_pref(jit_State *J, GCtrace *T, SnapEntry *map, MSize nmax,
438 BloomFilter seen, IRRef ref)
439{
440 IRIns *ir = &T->ir[ref];
441 TRef tr;
442 if (irref_isk(ref))
443 tr = snap_replay_const(J, ir);
444 else if (!regsp_used(ir->prev))
445 tr = 0;
446 else if (!bloomtest(seen, ref) || (tr = snap_dedup(J, map, nmax, ref)) == 0)
447 tr = emitir(IRT(IR_PVAL, irt_type(ir->t)), ref - REF_BIAS, 0);
448 return tr;
449}

451/* Check whether a sunk store corresponds to an allocation. Slow path. */
452static int snap_sunk_store2(GCtrace *T, IRIns *ira, IRIns *irs)
453{
454 if (irs->o == IR_ASTORE || irs->o == IR_HSTORE ||
455 irs->o == IR_FSTORE || irs->o == IR_XSTORE) {
456 IRIns *irk = &T->ir[irs->op1];
457 if (irk->o == IR_AREF || irk->o == IR_HREFK)
458 irk = &T->ir[irk->op1];
459 return (&T->ir[irk->op1] == ira);
460 }
461 return 0;
462}
463
464/* Check whether a sunk store corresponds to an allocation. Fast path. */
465static LJ_AINLINE int snap_sunk_store(GCtrace *T, IRIns *ira, IRIns *irs)
466{
467 if (irs->s != 255)
468 return (ira + irs->s == irs); /* Fast check. */
469 return snap_sunk_store2(T, ira, irs);
470}

472/* Replay snapshot state to setup side trace. */
473void lj_snap_replay(jit_State *J, GCtrace *T)
474{
475 SnapShot *snap = &T->snap[J->exitno];
476 SnapEntry *map = &T->snapmap[snap->mapofs];
477 MSize n, nent = snap->nent;
478 BloomFilter seen = 0;
479 int pass23 = 0;
480 J->framedepth = 0;
481 /* Emit IR for slots inherited from parent snapshot. */
482 for (n = 0; n < nent; n++) {
483 SnapEntry sn = map[n];
484 BCReg s = snap_slot(sn);
485 IRRef ref = snap_ref(sn);
486 IRIns *ir = &T->ir[ref];
487 TRef tr;
488 /* The bloom filter avoids O(nent^2) overhead for de-duping slots. */
489 if (bloomtest(seen, ref) && (tr = snap_dedup(J, map, n, ref)) != 0)
490 goto setslot;
491 bloomset(seen, ref);
492 if (irref_isk(ref)) {
493 /* See special treatment of LJ_FR2 slot 1 in snapshot_slots() above. */
494 if (LJ_FR2 && (sn == SNAP(1, SNAP_FRAME | SNAP_NORESTORE, REF_NIL)))
495 tr = 0;
496 else
497 tr = snap_replay_const(J, ir);
498 } else if (!regsp_used(ir->prev)) {
499 pass23 = 1;
500 lj_assertJ(s != 0, "unused slot 0 in snapshot");
501 tr = s;
502 } else {
503 IRType t = irt_type(ir->t);
504 uint32_t mode = IRSLOAD_INHERIT|IRSLOAD_PARENT;
505 if (LJ_SOFTFP32 && (sn & SNAP_SOFTFPNUM)) t = IRT_NUM;
506 if (ir->o == IR_SLOAD) mode |= (ir->op2 & IRSLOAD_READONLY);
507 tr = emitir_raw(IRT(IR_SLOAD, t), s, mode);
508 }
509 setslot:
510 J->slot[s] = tr | (sn&(SNAP_CONT|SNAP_FRAME)); /* Same as TREF_* flags. */
511 J->framedepth += ((sn & (SNAP_CONT|SNAP_FRAME)) && (s != LJ_FR2));
512 if ((sn & SNAP_FRAME))
513 J->baseslot = s+1;
514 }
515 if (pass23) {
516 IRIns *irlast = &T->ir[snap->ref];
517 pass23 = 0;
518 /* Emit dependent PVALs. */
519 for (n = 0; n < nent; n++) {
520 SnapEntry sn = map[n];
521 IRRef refp = snap_ref(sn);
522 IRIns *ir = &T->ir[refp];
523 if (regsp_reg(ir->r) == RID_SUNK) {
524 if (J->slot[snap_slot(sn)] != snap_slot(sn)) continue;
525 pass23 = 1;
526 lj_assertJ(ir->o == IR_TNEW || ir->o == IR_TDUP ||
527 ir->o == IR_CNEW || ir->o == IR_CNEWI,
528 "sunk parent IR %04d has bad op %d", refp - REF_BIAS, ir->o);
529 if (ir->op1 >= T->nk) snap_pref(J, T, map, nent, seen, ir->op1);
530 if (ir->op2 >= T->nk) snap_pref(J, T, map, nent, seen, ir->op2);
531 if (LJ_HASFFI && ir->o == IR_CNEWI) {
532 if (LJ_32 && refp+1 < T->nins && (ir+1)->o == IR_HIOP)
533 snap_pref(J, T, map, nent, seen, (ir+1)->op2);
534 } else {
535 IRIns *irs;
536 for (irs = ir+1; irs < irlast; irs++)
537 if (irs->r == RID_SINK && snap_sunk_store(T, ir, irs)) {
538 if (snap_pref(J, T, map, nent, seen, irs->op2) == 0)
539 snap_pref(J, T, map, nent, seen, T->ir[irs->op2].op1);
540 else if ((LJ_SOFTFP32 || (LJ_32 && LJ_HASFFI)) &&
541 irs+1 < irlast && (irs+1)->o == IR_HIOP)
542 snap_pref(J, T, map, nent, seen, (irs+1)->op2);
543 }
544 }
545 } else if (!irref_isk(refp) && !regsp_used(ir->prev)) {
546 lj_assertJ(ir->o == IR_CONV && ir->op2 == IRCONV_NUM_INT,
547 "sunk parent IR %04d has bad op %d", refp - REF_BIAS, ir->o);
548 J->slot[snap_slot(sn)] = snap_pref(J, T, map, nent, seen, ir->op1);
549 }
550 }
551 /* Replay sunk instructions. */
552 for (n = 0; pass23 && n < nent; n++) {
553 SnapEntry sn = map[n];
554 IRRef refp = snap_ref(sn);
555 IRIns *ir = &T->ir[refp];
556 if (regsp_reg(ir->r) == RID_SUNK) {
557 TRef op1, op2;
558 if (J->slot[snap_slot(sn)] != snap_slot(sn)) { /* De-dup allocs. */
559 J->slot[snap_slot(sn)] = J->slot[J->slot[snap_slot(sn)]];
560 continue;
561 }
562 op1 = ir->op1;
563 if (op1 >= T->nk) op1 = snap_pref(J, T, map, nent, seen, op1);
564 op2 = ir->op2;
565 if (op2 >= T->nk) op2 = snap_pref(J, T, map, nent, seen, op2);
566 if (LJ_HASFFI && ir->o == IR_CNEWI) {
567 if (LJ_32 && refp+1 < T->nins && (ir+1)->o == IR_HIOP) {
568 lj_needsplit(J); /* Emit joining HIOP. */
569 op2 = emitir_raw(IRT(IR_HIOP, IRT_I64), op2,
570 snap_pref(J, T, map, nent, seen, (ir+1)->op2));
571 }
572 J->slot[snap_slot(sn)] = emitir(ir->ot & ~(IRT_MARK|IRT_ISPHI), op1, op2);
573 } else {
574 IRIns *irs;
575 TRef tr = emitir(ir->ot, op1, op2);
576 J->slot[snap_slot(sn)] = tr;
577 for (irs = ir+1; irs < irlast; irs++)
578 if (irs->r == RID_SINK && snap_sunk_store(T, ir, irs)) {
579 IRIns *irr = &T->ir[irs->op1];
580 TRef val, key = irr->op2, tmp = tr;
581 if (irr->o != IR_FREF) {
582 IRIns *irk = &T->ir[key];
583 if (irr->o == IR_HREFK)
584 key = lj_ir_kslot(J, snap_replay_const(J, &T->ir[irk->op1]),
585 irk->op2);
586 else
587 key = snap_replay_const(J, irk);
588 if (irr->o == IR_HREFK || irr->o == IR_AREF) {
589 IRIns *irf = &T->ir[irr->op1];
590 tmp = emitir(irf->ot, tmp, irf->op2);
591 }
592 }
593 tmp = emitir(irr->ot, tmp, key);
594 val = snap_pref(J, T, map, nent, seen, irs->op2);
595 if (val == 0) {
596 IRIns *irc = &T->ir[irs->op2];
597 lj_assertJ(irc->o == IR_CONV && irc->op2 == IRCONV_NUM_INT,
598 "sunk store for parent IR %04d with bad op %d",
599 refp - REF_BIAS, irc->o);
600 val = snap_pref(J, T, map, nent, seen, irc->op1);
601 val = emitir(IRTN(IR_CONV), val, IRCONV_NUM_INT);
602 } else if ((LJ_SOFTFP32 || (LJ_32 && LJ_HASFFI)) &&
603 irs+1 < irlast && (irs+1)->o == IR_HIOP) {
604 IRType t = IRT_I64;
605 if (LJ_SOFTFP32 && irt_type((irs+1)->t) == IRT_SOFTFP)
606 t = IRT_NUM;
607 lj_needsplit(J);
608 if (irref_isk(irs->op2) && irref_isk((irs+1)->op2)) {
609 uint64_t k = (uint32_t)T->ir[irs->op2].i +
610 ((uint64_t)T->ir[(irs+1)->op2].i << 32);
611 val = lj_ir_k64(J, t == IRT_I64 ? IR_KINT64 : IR_KNUM, k);
612 } else {
613 val = emitir_raw(IRT(IR_HIOP, t), val,
614 snap_pref(J, T, map, nent, seen, (irs+1)->op2));
615 }
616 tmp = emitir(IRT(irs->o, t), tmp, val);
617 continue;
618 }
619 tmp = emitir(irs->ot, tmp, val);
620 } else if (LJ_HASFFI && irs->o == IR_XBAR && ir->o == IR_CNEW) {
621 emitir(IRT(IR_XBAR, IRT_NIL), 0, 0);
622 }
623 }
624 }
625 }
626 }
627 J->base = J->slot + J->baseslot;
628 J->maxslot = snap->nslots - J->baseslot;
629 lj_snap_add(J);
630 if (pass23) /* Need explicit GC step _after_ initial snapshot. */
631 emitir_raw(IRTG(IR_GCSTEP, IRT_NIL), 0, 0);
632}

640/* Restore a value from the trace exit state. */
641static void snap_restoreval(jit_State *J, GCtrace *T, ExitState *ex,
642 SnapNo snapno, BloomFilter rfilt,
643 IRRef ref, TValue *o)
644{
645 IRIns *ir = &T->ir[ref];
646 IRType1 t = ir->t;
647 RegSP rs = ir->prev;
648 if (irref_isk(ref)) { /* Restore constant slot. */
649 if (ir->o == IR_KPTR) {
650 o->u64 = (uint64_t)(uintptr_t)ir_kptr(ir);
651 } else {
652 lj_assertJ(!(ir->o == IR_KKPTR || ir->o == IR_KNULL),
653 "restore of const from IR %04d with bad op %d",
654 ref - REF_BIAS, ir->o);
655 lj_ir_kvalue(J->L, o, ir);
656 }
657 return;
658 }
659 if (LJ_UNLIKELY(bloomtest(rfilt, ref)))
660 rs = snap_renameref(T, snapno, ref, rs);
661 if (ra_hasspill(regsp_spill(rs))) { /* Restore from spill slot. */
662 int32_t *sps = &ex->spill[regsp_spill(rs)];
663 if (irt_isinteger(t)) {
664 setintV(o, *sps);
665#if !LJ_SOFTFP32
666 } else if (irt_isnum(t)) {
667 o->u64 = *(uint64_t *)sps;
668#endif
669#if LJ_64 && !LJ_GC64
670 } else if (irt_islightud(t)) {
671 /* 64 bit lightuserdata which may escape already has the tag bits. */
672 o->u64 = *(uint64_t *)sps;
673#endif
674 } else {
675 lj_assertJ(!irt_ispri(t), "PRI ref with spill slot");
676 setgcV(J->L, o, (GCobj *)(uintptr_t)*(GCSize *)sps, irt_toitype(t));
677 }
678 } else { /* Restore from register. */
679 Reg r = regsp_reg(rs);
680 if (ra_noreg(r)) {
681 lj_assertJ(ir->o == IR_CONV && ir->op2 == IRCONV_NUM_INT,
682 "restore from IR %04d has no reg", ref - REF_BIAS);
683 snap_restoreval(J, T, ex, snapno, rfilt, ir->op1, o);
684 if (LJ_DUALNUM) setnumV(o, (lua_Number)intV(o));
685 return;
686 } else if (irt_isinteger(t)) {
687 setintV(o, (int32_t)ex->gpr[r-RID_MIN_GPR]);
688#if !LJ_SOFTFP
689 } else if (irt_isnum(t)) {
690 setnumV(o, ex->fpr[r-RID_MIN_FPR]);
691#elif LJ_64 /* && LJ_SOFTFP */
692 } else if (irt_isnum(t)) {
693 o->u64 = ex->gpr[r-RID_MIN_GPR];
694#endif
695#if LJ_64 && !LJ_GC64
696 } else if (irt_is64(t)) {
697 /* 64 bit values that already have the tag bits. */
698 o->u64 = ex->gpr[r-RID_MIN_GPR];
699#endif
700 } else if (irt_ispri(t)) {
701 setpriV(o, irt_toitype(t));
702 } else {
703 setgcV(J->L, o, (GCobj *)ex->gpr[r-RID_MIN_GPR], irt_toitype(t));
704 }
705 }
706}

708#if LJ_HASFFI
709/* Restore raw data from the trace exit state. */
710static void snap_restoredata(jit_State *J, GCtrace *T, ExitState *ex,
711 SnapNo snapno, BloomFilter rfilt,
712 IRRef ref, void *dst, CTSize sz)
713{
714 IRIns *ir = &T->ir[ref];
715 RegSP rs = ir->prev;
716 int32_t *src;
717 uint64_t tmp;
718 UNUSED(J);
719 if (irref_isk(ref)) {
720 if (ir_isk64(ir)) {
721 src = (int32_t *)&ir[1];
722 } else if (sz == 8) {
723 tmp = (uint64_t)(uint32_t)ir->i;
724 src = (int32_t *)&tmp;
725 } else {
726 src = &ir->i;
727 }
728 } else {
729 if (LJ_UNLIKELY(bloomtest(rfilt, ref)))
730 rs = snap_renameref(T, snapno, ref, rs);
731 if (ra_hasspill(regsp_spill(rs))) {
732 src = &ex->spill[regsp_spill(rs)];
733 if (sz == 8 && !irt_is64(ir->t)) {
734 tmp = (uint64_t)(uint32_t)*src;
735 src = (int32_t *)&tmp;
736 }
737 } else {
738 Reg r = regsp_reg(rs);
739 if (ra_noreg(r)) {
740 /* Note: this assumes CNEWI is never used for SOFTFP split numbers. */
741 lj_assertJ(sz == 8 && ir->o == IR_CONV && ir->op2 == IRCONV_NUM_INT,
742 "restore from IR %04d has no reg", ref - REF_BIAS);
743 snap_restoredata(J, T, ex, snapno, rfilt, ir->op1, dst, 4);
744 *(lua_Number *)dst = (lua_Number)*(int32_t *)dst;
745 return;
746 }
747 src = (int32_t *)&ex->gpr[r-RID_MIN_GPR];
748#if !LJ_SOFTFP
749 if (r >= RID_MAX_GPR) {
750 src = (int32_t *)&ex->fpr[r-RID_MIN_FPR];
751#if LJ_TARGET_PPC
752 if (sz == 4) { /* PPC FPRs are always doubles. */
753 *(float *)dst = (float)*(double *)src;
754 return;
755 }
756#else
757 if (LJ_BE && sz == 4) src++;
758#endif
759 } else
760#endif
761 if (LJ_64 && LJ_BE && sz == 4) src++;
762 }
763 }
764 lj_assertJ(sz == 1 || sz == 2 || sz == 4 || sz == 8,
765 "restore from IR %04d with bad size %d", ref - REF_BIAS, sz);
766 if (sz == 4) *(int32_t *)dst = *src;
767 else if (sz == 8) *(int64_t *)dst = *(int64_t *)src;
768 else if (sz == 1) *(int8_t *)dst = (int8_t)*src;
769 else *(int16_t *)dst = (int16_t)*src;
770}

773/* Unsink allocation from the trace exit state. Unsink sunk stores. */
774static void snap_unsink(jit_State *J, GCtrace *T, ExitState *ex,
775 SnapNo snapno, BloomFilter rfilt,
776 IRIns *ir, TValue *o)
777{
778 lj_assertJ(ir->o == IR_TNEW || ir->o == IR_TDUP ||
779 ir->o == IR_CNEW || ir->o == IR_CNEWI,
780 "sunk allocation with bad op %d", ir->o);
781#if LJ_HASFFI
782 if (ir->o == IR_CNEW || ir->o == IR_CNEWI) {
783 CTState *cts = ctype_cts(J->L);
784 CTypeID id = (CTypeID)T->ir[ir->op1].i;
785 CTSize sz;
786 CTInfo info = lj_ctype_info(cts, id, &sz);
787 GCcdata *cd = lj_cdata_newx(cts, id, sz, info);
788 setcdataV(J->L, o, cd);
789 if (ir->o == IR_CNEWI) {
790 uint8_t *p = (uint8_t *)cdataptr(cd);
791 lj_assertJ(sz == 4 || sz == 8, "sunk cdata with bad size %d", sz);
792 if (LJ_32 && sz == 8 && ir+1 < T->ir + T->nins && (ir+1)->o == IR_HIOP) {
793 snap_restoredata(J, T, ex, snapno, rfilt, (ir+1)->op2,
794 LJ_LE ? p+4 : p, 4);
795 if (LJ_BE) p += 4;
796 sz = 4;
797 }
798 snap_restoredata(J, T, ex, snapno, rfilt, ir->op2, p, sz);
799 } else {
800 IRIns *irs, *irlast = &T->ir[T->snap[snapno].ref];
801 for (irs = ir+1; irs < irlast; irs++)
802 if (irs->r == RID_SINK && snap_sunk_store(T, ir, irs)) {
803 IRIns *iro = &T->ir[T->ir[irs->op1].op2];
804 uint8_t *p = (uint8_t *)cd;
805 CTSize szs;
806 lj_assertJ(irs->o == IR_XSTORE, "sunk store with bad op %d", irs->o);
807 lj_assertJ(T->ir[irs->op1].o == IR_ADD,
808 "sunk store with bad add op %d", T->ir[irs->op1].o);
809 lj_assertJ(iro->o == IR_KINT || iro->o == IR_KINT64,
810 "sunk store with bad const offset op %d", iro->o);
811 if (irt_is64(irs->t)) szs = 8;
812 else if (irt_isi8(irs->t) || irt_isu8(irs->t)) szs = 1;
813 else if (irt_isi16(irs->t) || irt_isu16(irs->t)) szs = 2;
814 else szs = 4;
815 if (LJ_64 && iro->o == IR_KINT64)
816 p += (int64_t)ir_k64(iro)->u64;
817 else
818 p += iro->i;
819 lj_assertJ(p >= (uint8_t *)cdataptr(cd) &&
820 p + szs <= (uint8_t *)cdataptr(cd) + sz,
821 "sunk store with offset out of range");
822 if (LJ_32 && irs+1 < T->ir + T->nins && (irs+1)->o == IR_HIOP) {
823 lj_assertJ(szs == 4, "sunk store with bad size %d", szs);
824 snap_restoredata(J, T, ex, snapno, rfilt, (irs+1)->op2,
825 LJ_LE ? p+4 : p, 4);
826 if (LJ_BE) p += 4;
827 }
828 snap_restoredata(J, T, ex, snapno, rfilt, irs->op2, p, szs);
829 }
830 }
831 } else
832#endif
833 {
834 IRIns *irs, *irlast;
835 GCtab *t = ir->o == IR_TNEW ? lj_tab_new(J->L, ir->op1, ir->op2) :
836 lj_tab_dup(J->L, ir_ktab(&T->ir[ir->op1]));
837 settabV(J->L, o, t);
838 irlast = &T->ir[T->snap[snapno].ref];
839 for (irs = ir+1; irs < irlast; irs++)
840 if (irs->r == RID_SINK && snap_sunk_store(T, ir, irs)) {
841 IRIns *irk = &T->ir[irs->op1];
842 TValue tmp, *val;
843 lj_assertJ(irs->o == IR_ASTORE || irs->o == IR_HSTORE ||
844 irs->o == IR_FSTORE,
845 "sunk store with bad op %d", irs->o);
846 if (irk->o == IR_FREF) {
847 lj_assertJ(irk->op2 == IRFL_TAB_META,
848 "sunk store with bad field %d", irk->op2);
849 snap_restoreval(J, T, ex, snapno, rfilt, irs->op2, &tmp);
850 /* NOBARRIER: The table is new (marked white). */
851 setgcref(t->metatable, obj2gco(tabV(&tmp)));
852 } else {
853 irk = &T->ir[irk->op2];
854 if (irk->o == IR_KSLOT) irk = &T->ir[irk->op1];
855 lj_ir_kvalue(J->L, &tmp, irk);
856 val = lj_tab_set(J->L, t, &tmp);
857 /* NOBARRIER: The table is new (marked white). */
858 snap_restoreval(J, T, ex, snapno, rfilt, irs->op2, val);
859 if (LJ_SOFTFP32 && irs+1 < T->ir + T->nins && (irs+1)->o == IR_HIOP) {
860 snap_restoreval(J, T, ex, snapno, rfilt, (irs+1)->op2, &tmp);
861 val->u32.hi = tmp.u32.lo;
862 }
863 }
864 }
865 }

868/* Restore interpreter state from exit state with the help of a snapshot. */
869const BCIns *lj_snap_restore(jit_State *J, void *exptr)
870{
871 ExitState *ex = (ExitState *)exptr;
872 SnapNo snapno = J->exitno; /* For now, snapno == exitno. */
873 GCtrace *T = traceref(J, J->parent);
874 SnapShot *snap = &T->snap[snapno];
875 MSize n, nent = snap->nent;
876 SnapEntry *map = &T->snapmap[snap->mapofs];
877#if !LJ_FR2 || defined(LUA_USE_ASSERT)
878 SnapEntry *flinks = &T->snapmap[snap_nextofs(T, snap)-1-LJ_FR2];
879#endif
880#if !LJ_FR2
881 ptrdiff_t ftsz0;
882#endif
883 TValue *frame;
884 BloomFilter rfilt = snap_renamefilter(T, snapno);
885 const BCIns *pc = snap_pc(&map[nent]);
886 lua_State *L = J->L;
887
888 /* Set interpreter PC to the next PC to get correct error messages. */
889 setcframe_pc(cframe_raw(L->cframe), pc+1);
890
891 /* Make sure the stack is big enough for the slots from the snapshot. */
892 if (LJ_UNLIKELY(L->base + snap->topslot >= tvref(L->maxstack))) {
893 L->top = curr_topL(L);
894 lj_state_growstack(L, snap->topslot - curr_proto(L)->framesize);
895 }
896
897 /* Fill stack slots with data from the registers and spill slots. */
898 frame = L->base-1-LJ_FR2;
899#if !LJ_FR2
900 ftsz0 = frame_ftsz(frame); /* Preserve link to previous frame in slot #0. */
901#endif
902 for (n = 0; n < nent; n++) {
903 SnapEntry sn = map[n];
904 if (!(sn & SNAP_NORESTORE)) {
905 TValue *o = &frame[snap_slot(sn)];
906 IRRef ref = snap_ref(sn);
907 IRIns *ir = &T->ir[ref];
908 if (ir->r == RID_SUNK) {
909 MSize j;
910 for (j = 0; j < n; j++)
911 if (snap_ref(map[j]) == ref) { /* De-duplicate sunk allocations. */
912 copyTV(L, o, &frame[snap_slot(map[j])]);
913 goto dupslot;
914 }
915 snap_unsink(J, T, ex, snapno, rfilt, ir, o);
916 dupslot:
917 continue;
918 }
919 snap_restoreval(J, T, ex, snapno, rfilt, ref, o);
920 if (LJ_SOFTFP32 && (sn & SNAP_SOFTFPNUM) && tvisint(o)) {
921 TValue tmp;
922 snap_restoreval(J, T, ex, snapno, rfilt, ref+1, &tmp);
923 o->u32.hi = tmp.u32.lo;
924#if !LJ_FR2
925 } else if ((sn & (SNAP_CONT|SNAP_FRAME))) {
926 /* Overwrite tag with frame link. */
927 setframe_ftsz(o, snap_slot(sn) != 0 ? (int32_t)*flinks-- : ftsz0);
928 L->base = o+1;
929#endif
930 }
931 }
932 }
933#if LJ_FR2
934 L->base += (map[nent+LJ_BE] & 0xff);
935#endif
936 lj_assertJ(map + nent == flinks, "inconsistent frames in snapshot");
937
938 /* Compute current stack top. */
939 switch (bc_op(*pc)) {
940 default:
941 if (bc_op(*pc) < BC_FUNCF) {
942 L->top = curr_topL(L);
943 break;
944 }
945 /* fallthrough */
946 case BC_CALLM: case BC_CALLMT: case BC_RETM: case BC_TSETM:
947 L->top = frame + snap->nslots;
948 break;
949 }
950 return pc;
951}
