34/* Check that the first argument is a string buffer. */
35static SBufExt *buffer_tobuf(lua_State *L)
36{
37 if (!(L->base < L->top && tvisbuf(L->base)))
38 lj_err_argtype(L, 1, "buffer");
39 return bufV(L->base);
40}
41
42/* Ditto, but for writers. */
43static LJ_AINLINE SBufExt *buffer_tobufw(lua_State *L)
44{
45 SBufExt *sbx = buffer_tobuf(L);
46 setsbufXL_(sbx, L);
47 return sbx;
48}
49
50#define buffer_toudata(sbx) ((GCudata *)(sbx)-1)

56LJLIB_CF(buffer_method_free)
57{
58 SBufExt *sbx = buffer_tobuf(L);
59 lj_bufx_free(L, sbx);
60 L->top = L->base+1; /* Chain buffer object. */

72LJLIB_CF(buffer_method_skip)
73{
74 SBufExt *sbx = buffer_tobuf(L);
75 MSize n = (MSize)lj_lib_checkintrange(L, 2, 0, LJ_MAX_BUF);
76 MSize len = sbufxlen(sbx);
77 if (n < len) {
78 sbx->r += n;
79 } else {
80 sbx->r = sbx->w = sbx->b;
81 }
82 L->top = L->base+1; /* Chain buffer object. */
83 return 1;
84}

86LJLIB_CF(buffer_method_set)
87{
88 SBufExt *sbx = buffer_tobuf(L);
89 GCobj *ref;
90 const char *p;
91 MSize len;
92#if LJ_HASFFI
93 if (tviscdata(L->base+1)) {
94 CTState *cts = ctype_cts(L);
95 lj_cconv_ct_tv(cts, ctype_get(cts, CTID_P_CVOID), (uint8_t *)&p,
96 L->base+1, CCF_ARG(2));
97 len = (MSize)lj_lib_checkintrange(L, 3, 0, LJ_MAX_BUF);
98 } else
99#endif
100 {
101 GCstr *str = lj_lib_checkstrx(L, 2);
102 p = strdata(str);
103 len = str->len;
104 }
105 lj_bufx_free(L, sbx);
106 lj_bufx_set_cow(L, sbx, p, len);
107 ref = gcV(L->base+1);
108 setgcref(sbx->cowref, ref);
109 lj_gc_objbarrier(L, buffer_toudata(sbx), ref);
110 L->top = L->base+1; /* Chain buffer object. */
111 return 1;
112}

114LJLIB_CF(buffer_method_put)
115{
116 SBufExt *sbx = buffer_tobufw(L);
117 ptrdiff_t arg, narg = L->top - L->base;
118 for (arg = 1; arg < narg; arg++) {
119 cTValue *o = &L->base[arg], *mo = NULL;
120 retry:
121 if (tvisstr(o)) {
122 lj_buf_putstr((SBuf *)sbx, strV(o));
123 } else if (tvisint(o)) {
124 lj_strfmt_putint((SBuf *)sbx, intV(o));
125 } else if (tvisnum(o)) {
126 lj_strfmt_putfnum((SBuf *)sbx, STRFMT_G14, numV(o));
127 } else if (tvisbuf(o)) {
128 SBufExt *sbx2 = bufV(o);
129 lj_buf_putmem((SBuf *)sbx, sbx2->r, sbufxlen(sbx2));
130 } else if (!mo && !tvisnil(mo = lj_meta_lookup(L, o, MM_tostring))) {
131 /* Call __tostring metamethod inline. */
132 copyTV(L, L->top++, mo);
133 copyTV(L, L->top++, o);
134 lua_call(L, 1, 1);
135 o = &L->base[arg]; /* The stack may have been reallocated. */
136 copyTV(L, &L->base[arg], L->top-1);
137 L->top = L->base + narg;
138 goto retry; /* Retry with the result. */
139 } else {
140 lj_err_argtype(L, arg+1, "string/number/__tostring");
141 }
142 /* Probably not useful to inline other __tostring MMs, e.g. FFI numbers. */
143 }
144 L->top = L->base+1; /* Chain buffer object. */
145 lj_gc_check(L);
146 return 1;
147}

158LJLIB_CF(buffer_method_get)
159{
160 SBufExt *sbx = buffer_tobuf(L);
161 ptrdiff_t arg, narg = L->top - L->base;
162 if (narg == 1) {
163 narg++;
164 setnilV(L->top++); /* get() is the same as get(nil). */
165 }
166 for (arg = 1; arg < narg; arg++) {
167 TValue *o = &L->base[arg];
168 MSize n = tvisnil(o) ? LJ_MAX_BUF :
169 (MSize) lj_lib_checkintrange(L, arg+1, 0, LJ_MAX_BUF);
170 MSize len = sbufxlen(sbx);
171 if (n > len) n = len;
172 setstrV(L, o, lj_str_new(L, sbx->r, n));
173 sbx->r += n;
174 }
175 if (sbx->r == sbx->w) sbx->r = sbx->w = sbx->b;
176 lj_gc_check(L);
177 return narg-1;
178}

181LJLIB_CF(buffer_method_putcdata)
182{
183 SBufExt *sbx = buffer_tobufw(L);
184 const char *p;
185 MSize len;
186 if (tviscdata(L->base+1)) {
187 CTState *cts = ctype_cts(L);
188 lj_cconv_ct_tv(cts, ctype_get(cts, CTID_P_CVOID), (uint8_t *)&p,
189 L->base+1, CCF_ARG(2));
190 } else {
191 lj_err_argtype(L, 2, "cdata");
192 }
193 len = (MSize)lj_lib_checkintrange(L, 3, 0, LJ_MAX_BUF);
194 lj_buf_putmem((SBuf *)sbx, p, len);
195 L->top = L->base+1; /* Chain buffer object. */
196 return 1;

199LJLIB_CF(buffer_method_reserve)
200{
201 SBufExt *sbx = buffer_tobufw(L);
202 MSize len = (MSize)lj_lib_checkintrange(L, 2, 0, LJ_MAX_BUF);
203 GCcdata *cd;
204 lj_buf_more((SBuf *)sbx, len);
205 ctype_loadffi(L);
206 cd = lj_cdata_new_(L, CTID_P_UINT8, CTSIZE_PTR);
207 *(void **)cdataptr(cd) = sbx->w;
208 setcdataV(L, L->top++, cd);
209 setintV(L->top++, sbufleft(sbx));
210 return 2;
211}

213LJLIB_CF(buffer_method_commit)
214{
215 SBufExt *sbx = buffer_tobuf(L);
216 MSize len = (MSize)lj_lib_checkintrange(L, 2, 0, LJ_MAX_BUF);
217 if (len > sbufleft(sbx)) lj_err_arg(L, 2, LJ_ERR_NUMRNG);
218 sbx->w += len;
219 L->top = L->base+1; /* Chain buffer object. */
220 return 1;
221}

286LJLIB_CF(buffer_new)
287{
288 MSize sz = 0;
289 int targ = 1;
290 GCtab *env, *dict = NULL;
291 GCudata *ud;
292 SBufExt *sbx;
293 if (L->base < L->top && !tvistab(L->base)) {
294 targ = 2;
295 if (!tvisnil(L->base))
296 sz = (MSize)lj_lib_checkintrange(L, 1, 0, LJ_MAX_BUF);
297 }
298 if (L->base+targ-1 < L->top) {
299 GCtab *options = lj_lib_checktab(L, targ);
300 cTValue *opt_dict = lj_tab_getstr(options, lj_str_newlit(L, "dict"));
301 if (opt_dict && tvistab(opt_dict)) {
302 dict = tabV(opt_dict);
303 lj_serialize_dict_prep(L, dict);
304 }
305 }
306 env = tabref(curr_func(L)->c.env);
307 ud = lj_udata_new(L, sizeof(SBufExt), env);
308 ud->udtype = UDTYPE_BUFFER;
309 /* NOBARRIER: The GCudata is new (marked white). */
310 setgcref(ud->metatable, obj2gco(env));
311 setudataV(L, L->top++, ud);
312 sbx = (SBufExt *)uddata(ud);
313 lj_bufx_init(L, sbx);
314 setgcref(sbx->dict, obj2gco(dict));
315 if (sz > 0) lj_buf_need2((SBuf *)sbx, sz);
316 return 1;
317}

319LJLIB_CF(buffer_encode)
320{
321 cTValue *o = lj_lib_checkany(L, 1);
322 SBufExt sbx;
323 memset(&sbx, 0, sizeof(SBufExt));
324 lj_bufx_set_borrow(L, &sbx, &G(L)->tmpbuf);
325 lj_serialize_put(&sbx, o);
326 setstrV(L, L->top++, lj_buf_str(L, (SBuf *)&sbx));
327 lj_gc_check(L);
328 return 1;
329}
330
331LJLIB_CF(buffer_decode)
332{
333 GCstr *str = lj_lib_checkstrx(L, 1);
334 SBufExt sbx;
335 memset(&sbx, 0, sizeof(SBufExt));
336 lj_bufx_set_cow(L, &sbx, strdata(str), str->len);
337 /* No need to set sbx.cowref here. */
338 setnilV(L->top++);
339 lj_serialize_get(&sbx, L->top-1);
340 lj_gc_check(L);
341 if (sbx.r != sbx.w) lj_err_caller(L, LJ_ERR_BUFFER_LEFTOV);
342 return 1;
343}

349int luaopen_string_buffer(lua_State *L)
350{
351 LJ_LIB_REG(L, NULL, buffer_method);
352 lua_getfield(L, -1, "__tostring");
353 lua_setfield(L, -2, "tostring");
354 LJ_LIB_REG(L, NULL, buffer);
355 return 1;
356}
