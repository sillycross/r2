260#define lua_register(L,n,f) (lua_pushcfunction(L, (f)), lua_setglobal(L, (n)))

264#define lua_strlen(L,i) lua_objlen(L, (i))
265
266#define lua_isfunction(L,n) (lua_type(L, (n)) == LUA_TFUNCTION)
267#define lua_istable(L,n) (lua_type(L, (n)) == LUA_TTABLE)
268#define lua_islightuserdata(L,n) (lua_type(L, (n)) == LUA_TLIGHTUSERDATA)
269#define lua_isnil(L,n) (lua_type(L, (n)) == LUA_TNIL)
270#define lua_isboolean(L,n) (lua_type(L, (n)) == LUA_TBOOLEAN)
271#define lua_isthread(L,n) (lua_type(L, (n)) == LUA_TTHREAD)
272#define lua_isnone(L,n) (lua_type(L, (n)) == LUA_TNONE)
273#define lua_isnoneornil(L, n) (lua_type(L, (n)) <= 0)

360struct lua_Debug {
361 int event;
362 const char *name; /* (n) */
363 const char *namewhat; /* (n) `global', `local', `field', `method' */
364 const char *what; /* (S) `Lua', `C', `main', `tail' */
365 const char *source; /* (S) */
366 int currentline; /* (l) */
367 int nups; /* (u) number of upvalues */
368 int linedefined; /* (S) */
369 int lastlinedefined; /* (S) */
370 char short_src[LUA_IDSIZE]; /* (S) */
371 /* private part */
372 int i_ci; /* active function */
373};
