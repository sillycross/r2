23/* Tags for internal serialization format. */
24enum {
25 SER_TAG_NIL, /* 0x00 */
26 SER_TAG_FALSE,
27 SER_TAG_TRUE,
28 SER_TAG_NULL,
29 SER_TAG_LIGHTUD32,
30 SER_TAG_LIGHTUD64,
31 SER_TAG_INT,
32 SER_TAG_NUM,
33 SER_TAG_TAB, /* 0x08 */
34 SER_TAG_0x0e = SER_TAG_TAB+6,
35 SER_TAG_DICT,
36 SER_TAG_INT64, /* 0x10 */
37 SER_TAG_UINT64,
38 SER_TAG_COMPLEX,
39 SER_TAG_0x13,
40 SER_TAG_0x14,
41 SER_TAG_0x15,
42 SER_TAG_0x16,
43 SER_TAG_0x17,
44 SER_TAG_0x18, /* 0x18 */
45 SER_TAG_0x19,
46 SER_TAG_0x1a,
47 SER_TAG_0x1b,
48 SER_TAG_0x1c,
49 SER_TAG_0x1d,
50 SER_TAG_0x1e,
51 SER_TAG_0x1f,
52 SER_TAG_STR, /* 0x20 + str->len */
53};
54LJ_STATIC_ASSERT((SER_TAG_TAB & 7) == 0);

58static LJ_AINLINE char *serialize_more(char *w, SBufExt *sbx, MSize sz)
59{
60 if (LJ_UNLIKELY(sz > (MSize)(sbx->e - w))) {
61 sbx->w = w;
62 w = lj_buf_more2((SBuf *)sbx, sz);
63 }
64 return w;
65}

67/* Write U124 to buffer. */
68static LJ_NOINLINE char *serialize_wu124_(char *w, uint32_t v)
69{
70 if (v < 0x1fe0) {
71 v -= 0xe0;
72 *w++ = (char)(0xe0 | (v >> 8)); *w++ = (char)v;
73 } else {
74 *w++ = (char)0xff;
75#if LJ_BE
76 v = lj_bswap(v);
77#endif
78 memcpy(w, &v, 4); w += 4;
79 }
80 return w;
81}
82
83static LJ_AINLINE char *serialize_wu124(char *w, uint32_t v)
84{
85 if (LJ_LIKELY(v < 0xe0)) {
86 *w++ = (char)v;
87 return w;
88 } else {
89 return serialize_wu124_(w, v);
90 }
91}
92
93static LJ_NOINLINE char *serialize_ru124_(char *r, char *w, uint32_t *pv)
94{
95 uint32_t v = *pv;
96 if (v != 0xff) {
97 if (r >= w) return NULL;
98 v = ((v & 0x1f) << 8) + *(uint8_t *)r + 0xe0; r++;
99 } else {
100 if (r + 4 > w) return NULL;
101 v = lj_getu32(r); r += 4;
102#if LJ_BE
103 v = lj_bswap(v);
104#endif
105 }
106 *pv = v;
107 return r;
108}
109
110static LJ_AINLINE char *serialize_ru124(char *r, char *w, uint32_t *pv)
111{
112 if (LJ_LIKELY(r < w)) {
113 uint32_t v = *(uint8_t *)r; r++;
114 *pv = v;
115 if (LJ_UNLIKELY(v >= 0xe0)) {
116 r = serialize_ru124_(r, w, pv);
117 }
118 return r;
119 }
120 return NULL;
121}

123/* Prepare string dictionary for use (once). */
124void LJ_FASTCALL lj_serialize_dict_prep(lua_State *L, GCtab *dict)
125{
126 if (!dict->hmask) { /* No hash part means not prepared, yet. */
127 MSize i, len = lj_tab_len(dict);
128 if (!len) return;
129 lj_tab_resize(L, dict, dict->asize, hsize2hbits(len));
130 for (i = 1; i <= len && i < dict->asize; i++) {
131 cTValue *o = arrayslot(dict, i);
132 if (tvisstr(o)) {
133 if (!lj_tab_getstr(dict, strV(o))) { /* Ignore dups. */
134 lj_tab_newkey(L, dict, o)->u64 = (uint64_t)(i-1);
135 }
136 } else if (!tvisfalse(o)) {
137 lj_err_caller(L, LJ_ERR_BUFFER_BADOPT);
138 }
139 }
140 }
141}

145/* Put serialized object into buffer. */
146static char *serialize_put(char *w, SBufExt *sbx, cTValue *o)
147{
148 if (LJ_LIKELY(tvisstr(o))) {
149 const GCstr *str = strV(o);
150 MSize len = str->len;
151 w = serialize_more(w, sbx, 5+len);
152 w = serialize_wu124(w, SER_TAG_STR + len);
153 w = lj_buf_wmem(w, strdata(str), len);
154 } else if (tvisint(o)) {
155 uint32_t x = LJ_BE ? lj_bswap((uint32_t)intV(o)) : (uint32_t)intV(o);
156 w = serialize_more(w, sbx, 1+4);
157 *w++ = SER_TAG_INT; memcpy(w, &x, 4); w += 4;
158 } else if (tvisnum(o)) {
159 uint64_t x = LJ_BE ? lj_bswap64(o->u64) : o->u64;
160 w = serialize_more(w, sbx, 1+sizeof(lua_Number));
161 *w++ = SER_TAG_NUM; memcpy(w, &x, 8); w += 8;
162 } else if (tvispri(o)) {
163 w = serialize_more(w, sbx, 1);
164 *w++ = (char)(SER_TAG_NIL + ~itype(o));
165 } else if (tvistab(o)) {
166 const GCtab *t = tabV(o);
167 uint32_t narray = 0, nhash = 0, one = 2;
168 if (sbx->depth <= 0) lj_err_caller(sbufL(sbx), LJ_ERR_BUFFER_DEPTH);
169 sbx->depth--;
170 if (t->asize > 0) { /* Determine max. length of array part. */
171 ptrdiff_t i;
172 TValue *array = tvref(t->array);
173 for (i = (ptrdiff_t)t->asize-1; i >= 0; i--)
174 if (!tvisnil(&array[i]))
175 break;
176 narray = (uint32_t)(i+1);
177 if (narray && tvisnil(&array[0])) one = 4;
178 }
179 if (t->hmask > 0) { /* Count number of used hash slots. */
180 uint32_t i, hmask = t->hmask;
181 Node *node = noderef(t->node);
182 for (i = 0; i <= hmask; i++)
183 nhash += !tvisnil(&node[i].val);
184 }
185 /* Write number of array slots and hash slots. */
186 w = serialize_more(w, sbx, 1+2*5);
187 *w++ = (char)(SER_TAG_TAB + (nhash ? 1 : 0) + (narray ? one : 0));
188 if (narray) w = serialize_wu124(w, narray);
189 if (nhash) w = serialize_wu124(w, nhash);
190 if (narray) { /* Write array entries. */
191 cTValue *oa = tvref(t->array) + (one >> 2);
192 cTValue *oe = tvref(t->array) + narray;
193 while (oa < oe) w = serialize_put(w, sbx, oa++);
194 }
195 if (nhash) { /* Write hash entries. */
196 const Node *node = noderef(t->node) + t->hmask;
197 GCtab *dict = tabref(sbx->dict);
198 if (LJ_UNLIKELY(dict)) {
199 for (;; node--)
200 if (!tvisnil(&node->val)) {
201 if (LJ_LIKELY(tvisstr(&node->key))) {
202 /* Inlined lj_tab_getstr is 30% faster. */
203 const GCstr *str = strV(&node->key);
204 Node *n = hashstr(dict, str);
205 do {
206 if (tvisstr(&n->key) && strV(&n->key) == str) {
207 uint32_t idx = n->val.u32.lo;
208 w = serialize_more(w, sbx, 1+5);
209 *w++ = SER_TAG_DICT;
210 w = serialize_wu124(w, idx);
211 break;
212 }
213 n = nextnode(n);
214 if (!n) {
215 MSize len = str->len;
216 w = serialize_more(w, sbx, 5+len);
217 w = serialize_wu124(w, SER_TAG_STR + len);
218 w = lj_buf_wmem(w, strdata(str), len);
219 break;
220 }
221 } while (1);
222 } else {
223 w = serialize_put(w, sbx, &node->key);
224 }
225 w = serialize_put(w, sbx, &node->val);
226 if (--nhash == 0) break;
227 }
228 } else {
229 for (;; node--)
230 if (!tvisnil(&node->val)) {
231 w = serialize_put(w, sbx, &node->key);
232 w = serialize_put(w, sbx, &node->val);
233 if (--nhash == 0) break;
234 }
235 }
236 }
237 sbx->depth++;
238#if LJ_HASFFI
239 } else if (tviscdata(o)) {
240 CTState *cts = ctype_cts(sbufL(sbx));
241 CType *s = ctype_raw(cts, cdataV(o)->ctypeid);
242 uint8_t *sp = cdataptr(cdataV(o));
243 if (ctype_isinteger(s->info) && s->size == 8) {
244 w = serialize_more(w, sbx, 1+8);
245 *w++ = (s->info & CTF_UNSIGNED) ? SER_TAG_UINT64 : SER_TAG_INT64;
246#if LJ_BE
247 { uint64_t u = lj_bswap64(*(uint64_t *)sp); memcpy(w, &u, 8); }
248#else
249 memcpy(w, sp, 8);
250#endif
251 w += 8;
252 } else if (ctype_iscomplex(s->info) && s->size == 16) {
253 w = serialize_more(w, sbx, 1+16);
254 *w++ = SER_TAG_COMPLEX;
255#if LJ_BE
256 { /* Only swap the doubles. The re/im order stays the same. */
257 uint64_t u = lj_bswap64(((uint64_t *)sp)[0]); memcpy(w, &u, 8);
258 u = lj_bswap64(((uint64_t *)sp)[1]); memcpy(w+8, &u, 8);
259 }
260#else
261 memcpy(w, sp, 16);
262#endif
263 w += 16;
264 } else {
265 goto badenc; /* NYI other cdata */
266 }
267#endif
268 } else if (tvislightud(o)) {
269 uintptr_t ud = (uintptr_t)lightudV(G(sbufL(sbx)), o);
270 w = serialize_more(w, sbx, 1+sizeof(ud));
271 if (ud == 0) {
272 *w++ = SER_TAG_NULL;
273 } else if (LJ_32 || checku32(ud)) {
274#if LJ_BE && LJ_64
275 ud = lj_bswap64(ud);
276#elif LJ_BE
277 ud = lj_bswap(ud);
278#endif
279 *w++ = SER_TAG_LIGHTUD32; memcpy(w, &ud, 4); w += 4;
280#if LJ_64
281 } else {
282#if LJ_BE
283 ud = lj_bswap64(ud);
284#endif
285 *w++ = SER_TAG_LIGHTUD64; memcpy(w, &ud, 8); w += 8;
286#endif
287 }
288 } else {
289 /* NYI userdata */
290#if LJ_HASFFI
291 badenc:
292#endif
293 lj_err_callerv(sbufL(sbx), LJ_ERR_BUFFER_BADENC, lj_typename(o));
294 }
295 return w;
296}

298/* Get serialized object from buffer. */
299static char *serialize_get(char *r, SBufExt *sbx, TValue *o)
300{
301 char *w = sbx->w;
302 uint32_t tp;
303 r = serialize_ru124(r, w, &tp); if (LJ_UNLIKELY(!r)) goto eob;
304 if (LJ_LIKELY(tp >= SER_TAG_STR)) {
305 uint32_t len = tp - SER_TAG_STR;
306 if (LJ_UNLIKELY(len > (uint32_t)(w - r))) goto eob;
307 setstrV(sbufL(sbx), o, lj_str_new(sbufL(sbx), r, len));
308 r += len;
309 } else if (tp == SER_TAG_INT) {
310 if (LJ_UNLIKELY(r + 4 > w)) goto eob;
311 setintV(o, (int32_t)(LJ_BE ? lj_bswap(lj_getu32(r)) : lj_getu32(r)));
312 r += 4;
313 } else if (tp == SER_TAG_NUM) {
314 if (LJ_UNLIKELY(r + 8 > w)) goto eob;
315 memcpy(o, r, 8); r += 8;
316#if LJ_BE
317 o->u64 = lj_bswap64(o->u64);
318#endif
319 if (!tvisnum(o)) setnanV(o);
320 } else if (tp <= SER_TAG_TRUE) {
321 setpriV(o, ~tp);
322 } else if (tp == SER_TAG_DICT) {
323 GCtab *dict;
324 uint32_t idx;
325 r = serialize_ru124(r, w, &idx);
326 idx++;
327 dict = tabref(sbx->dict);
328 if (dict && idx < dict->asize && tvisstr(arrayslot(dict, idx)))
329 copyTV(sbufL(sbx), o, arrayslot(dict, idx));
330 else
331 lj_err_callerv(sbufL(sbx), LJ_ERR_BUFFER_BADDICTX, idx);
332 } else if (tp >= SER_TAG_TAB && tp < SER_TAG_TAB+6) {
333 uint32_t narray = 0, nhash = 0;
334 GCtab *t;
335 if (tp >= SER_TAG_TAB+2) {
336 r = serialize_ru124(r, w, &narray); if (LJ_UNLIKELY(!r)) goto eob;
337 }
338 if ((tp & 1)) {
339 r = serialize_ru124(r, w, &nhash); if (LJ_UNLIKELY(!r)) goto eob;
340 }
341 t = lj_tab_new(sbufL(sbx), narray, hsize2hbits(nhash));
342 settabV(sbufL(sbx), o, t);
343 if (narray) {
344 TValue *oa = tvref(t->array) + (tp >= SER_TAG_TAB+4);
345 TValue *oe = tvref(t->array) + narray;
346 while (oa < oe) r = serialize_get(r, sbx, oa++);
347 }
348 if (nhash) {
349 do {
350 TValue k, *v;
351 r = serialize_get(r, sbx, &k);
352 v = lj_tab_set(sbufL(sbx), t, &k);
353 if (LJ_UNLIKELY(!tvisnil(v)))
354 lj_err_caller(sbufL(sbx), LJ_ERR_BUFFER_DUPKEY);
355 r = serialize_get(r, sbx, v);
356 } while (--nhash);
357 }
358#if LJ_HASFFI
359 } else if (tp >= SER_TAG_INT64 && tp <= SER_TAG_COMPLEX) {
360 uint32_t sz = tp == SER_TAG_COMPLEX ? 16 : 8;
361 GCcdata *cd;
362 if (LJ_UNLIKELY(r + sz > w)) goto eob;
363 cd = lj_cdata_new_(sbufL(sbx),
364 tp == SER_TAG_INT64 ? CTID_INT64 :
365 tp == SER_TAG_UINT64 ? CTID_UINT64 : CTID_COMPLEX_DOUBLE,
366 sz);
367 memcpy(cdataptr(cd), r, sz); r += sz;
368#if LJ_BE
369 *(uint64_t *)cdataptr(cd) = lj_bswap64(*(uint64_t *)cdataptr(cd));
370 if (sz == 16)
371 ((uint64_t *)cdataptr(cd))[1] = lj_bswap64(((uint64_t *)cdataptr(cd))[1]);
372#endif
373 setcdataV(sbufL(sbx), o, cd);
374#endif
375 } else if (tp <= (LJ_64 ? SER_TAG_LIGHTUD64 : SER_TAG_LIGHTUD32)) {
376 uintptr_t ud = 0;
377 if (tp == SER_TAG_LIGHTUD32) {
378 if (LJ_UNLIKELY(r + 4 > w)) goto eob;
379 ud = (uintptr_t)(LJ_BE ? lj_bswap(lj_getu32(r)) : lj_getu32(r));
380 r += 4;
381 }
382#if LJ_64
383 else if (tp == SER_TAG_LIGHTUD64) {
384 if (LJ_UNLIKELY(r + 8 > w)) goto eob;
385 memcpy(&ud, r, 8); r += 8;
386#if LJ_BE
387 ud = lj_bswap64(ud);
388#endif
389 }
390 setrawlightudV(o, lj_lightud_intern(sbufL(sbx), (void *)ud));
391#else
392 setrawlightudV(o, (void *)ud);
393#endif
394 } else {
395 lj_err_callerv(sbufL(sbx), LJ_ERR_BUFFER_BADDEC, tp);
396 }
397 return r;
398eob:
399 lj_err_caller(sbufL(sbx), LJ_ERR_BUFFER_EOB);
400 return NULL;
401}

403SBufExt * LJ_FASTCALL lj_serialize_put(SBufExt *sbx, cTValue *o)
404{
405 sbx->depth = LJ_SERIALIZE_DEPTH;
406 sbx->w = serialize_put(sbx->w, sbx, o);
407 return sbx;
408}

410SBufExt * LJ_FASTCALL lj_serialize_get(SBufExt *sbx, TValue *o)
411{
412 sbx->r = serialize_get(sbx->r, sbx, o);
413 return sbx;
