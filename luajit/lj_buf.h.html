13/* Resizable string buffers. */
14
15/* The SBuf struct definition is in lj_obj.h:
16** char *w; Write pointer.
17** char *e; End pointer.
18** char *b; Base pointer.
19** MRef L; lua_State, used for buffer resizing. Extension bits in 3 LSB.
20*/
21
22/* Extended string buffer. */
23typedef struct SBufExt {
24 SBufHeader;
25 union {
26 GCRef cowref; /* Copy-on-write object reference. */
27 MRef bsb; /* Borrowed string buffer. */
28 };
29 char *r; /* Read pointer. */
30 GCRef dict; /* Serialization string dictionary table. */
31 int depth; /* Remaining recursion depth. */
32} SBufExt;
33
34#define sbufsz(sb) ((MSize)((sb)->e - (sb)->b))
35#define sbuflen(sb) ((MSize)((sb)->w - (sb)->b))
36#define sbufleft(sb) ((MSize)((sb)->e - (sb)->w))
37#define sbufxlen(sbx) ((MSize)((sbx)->w - (sbx)->r))
38#define sbufxslack(sbx) ((MSize)((sbx)->r - (sbx)->b))
39
40#define SBUF_MASK_FLAG (7)
41#define SBUF_MASK_L (~(GCSize)SBUF_MASK_FLAG)
42#define SBUF_FLAG_EXT 1 /* Extended string buffer. */
43#define SBUF_FLAG_COW 2 /* Copy-on-write buffer. */
44#define SBUF_FLAG_BORROW 4 /* Borrowed string buffer. */
45
46#define sbufL(sb) \
47 ((lua_State *)(void *)(uintptr_t)(mrefu((sb)->L) & SBUF_MASK_L))
48#define setsbufL(sb, l) (setmref((sb)->L, (l)))
49#define setsbufXL(sb, l, flag) \
50 (setmrefu((sb)->L, (GCSize)(uintptr_t)(void *)(l) + (flag)))
51#define setsbufXL_(sb, l) \
52 (setmrefu((sb)->L, (GCSize)(uintptr_t)(void *)(l) | (mrefu((sb)->L) & SBUF_MASK_FLAG)))
53
54#define sbufflag(sb) (mrefu((sb)->L))
55#define sbufisext(sb) (sbufflag((sb)) & SBUF_FLAG_EXT)
56#define sbufiscow(sb) (sbufflag((sb)) & SBUF_FLAG_COW)
57#define sbufisborrow(sb) (sbufflag((sb)) & SBUF_FLAG_BORROW)
58#define sbufX(sb) \
59 (lj_assertG_(G(sbufL(sb)), sbufisext(sb), "not an SBufExt"), (SBufExt *)(sb))
60#define setsbufflag(sb, flag) (setmrefu((sb)->L, (flag)))
61
62#define tvisbuf(o) \
63 (LJ_HASBUFFER && tvisudata(o) && udataV(o)->udtype == UDTYPE_BUFFER)
64#define bufV(o) check_exp(tvisbuf(o), ((SBufExt *)uddata(udataV(o))))

72static LJ_AINLINE void lj_buf_init(lua_State *L, SBuf *sb)
73{
74 setsbufL(sb, L);
75 sb->w = sb->e = sb->b = NULL;
76}
77
78static LJ_AINLINE void lj_buf_reset(SBuf *sb)
79{
80 sb->w = sb->b;
81}
82
83static LJ_AINLINE SBuf *lj_buf_tmp_(lua_State *L)
84{
85 SBuf *sb = &G(L)->tmpbuf;
86 setsbufL(sb, L);
87 lj_buf_reset(sb);
88 return sb;
89}
90
91static LJ_AINLINE void lj_buf_free(global_State *g, SBuf *sb)
92{
93 lj_assertG(!sbufisext(sb), "bad free of SBufExt");
94 lj_mem_free(g, sb->b, sbufsz(sb));
95}
96
97static LJ_AINLINE char *lj_buf_need(SBuf *sb, MSize sz)
98{
99 if (LJ_UNLIKELY(sz > sbufsz(sb)))
100 return lj_buf_need2(sb, sz);
101 return sb->b;
102}
103
104static LJ_AINLINE char *lj_buf_more(SBuf *sb, MSize sz)
105{
106 if (LJ_UNLIKELY(sz > sbufleft(sb)))
107 return lj_buf_more2(sb, sz);
108 return sb->w;
109}

111/* Extended buffer management */
112static LJ_AINLINE void lj_bufx_init(lua_State *L, SBufExt *sbx)
113{
114 memset(sbx, 0, sizeof(SBufExt));
115 setsbufXL(sbx, L, SBUF_FLAG_EXT);
116}
117
118static LJ_AINLINE void lj_bufx_set_borrow(lua_State *L, SBufExt *sbx, SBuf *sb)
119{
120 setsbufXL(sbx, L, SBUF_FLAG_EXT | SBUF_FLAG_BORROW);
121 setmref(sbx->bsb, sb);
122 sbx->r = sbx->w = sbx->b = sb->b;
123 sbx->e = sb->e;
124}
125
126static LJ_AINLINE void lj_bufx_set_cow(lua_State *L, SBufExt *sbx,
127 const char *p, MSize len)
128{
129 setsbufXL(sbx, L, SBUF_FLAG_EXT | SBUF_FLAG_COW);
130 sbx->r = sbx->b = (char *)p;
131 sbx->w = sbx->e = (char *)p + len;
132}
133
134static LJ_AINLINE void lj_bufx_reset(SBufExt *sbx)
135{
136 if (sbufiscow(sbx)) {
137 setmrefu(sbx->L, (mrefu(sbx->L) & ~(GCSize)SBUF_FLAG_COW));
138 setgcrefnull(sbx->cowref);
139 sbx->b = sbx->e = NULL;
140 }
141 sbx->r = sbx->w = sbx->b;
142}
143
144static LJ_AINLINE void lj_bufx_free(lua_State *L, SBufExt *sbx)
145{
146 if (!sbufiscow(sbx)) lj_mem_free(G(L), sbx->b, sbufsz(sbx));
147 setsbufXL(sbx, L, SBUF_FLAG_EXT);
148 setgcrefnull(sbx->cowref);
149 sbx->r = sbx->w = sbx->b = sbx->e = NULL;
150}

162static LJ_AINLINE void lj_buf_putb(SBuf *sb, int c)
163{
164 char *w = lj_buf_more(sb, 1);
165 *w++ = (char)c;
166 sb->w = w;
167}
