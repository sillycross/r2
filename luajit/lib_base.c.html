87{

89 cTValue *mo = lj_meta_lookup(L, o, mm);

98 if (mm == MM_pairs) setnilV(o+1); else setintV(o+1, 0);
99 return FFH_RES(3);
100 }
101}

142LJLIB_CF(getfenv) LJLIB_REC(.)
143{
144 GCfunc *fn;
145 cTValue *o = L->base;
146 if (!(o < L->top && tvisfunc(o))) {
147 int level = lj_lib_optint(L, 1, 1);
148 o = lj_debug_frame(L, level, &level);
149 if (o == NULL)
150 lj_err_arg(L, 1, LJ_ERR_INVLVL);
151 if (LJ_FR2) o--;
152 }
153 fn = &gcval(o)->fn;
154 settabV(L, L->top++, isluafunc(fn) ? tabref(fn->l.env) : tabref(L->env));
155 return 1;
156}

158LJLIB_CF(setfenv)
159{
160 GCfunc *fn;
161 GCtab *t = lj_lib_checktab(L, 2);
162 cTValue *o = L->base;
163 if (!(o < L->top && tvisfunc(o))) {
164 int level = lj_lib_checkint(L, 1);
165 if (level == 0) {
166 /* NOBARRIER: A thread (i.e. L) is never black. */
167 setgcref(L->env, obj2gco(t));
168 return 0;
169 }
170 o = lj_debug_frame(L, level, &level);
171 if (o == NULL)
172 lj_err_arg(L, 1, LJ_ERR_INVLVL);
173 if (LJ_FR2) o--;
174 }
175 fn = &gcval(o)->fn;
176 if (!isluafunc(fn))
177 lj_err_caller(L, LJ_ERR_SETFENV);
178 setgcref(fn->l.env, obj2gco(t));
179 lj_gc_objbarrier(L, obj2gco(fn), t);
180 setfuncV(L, L->top++, fn);
181 return 1;
182}

222LJLIB_CF(unpack)
223{
224 GCtab *t = lj_lib_checktab(L, 1);
225 int32_t n, i = lj_lib_optint(L, 2, 1);
226 int32_t e = (L->base+3-1 < L->top && !tvisnil(L->base+3-1)) ?
227 lj_lib_checkint(L, 3) : (int32_t)lj_tab_len(t);
228 uint32_t nu;
229 if (i > e) return 0;
230 nu = (uint32_t)e - (uint32_t)i;
231 n = (int32_t)(nu+1);
232 if (nu >= LUAI_MAXCSTACK || !lua_checkstack(L, n))
233 lj_err_caller(L, LJ_ERR_UNPACK);
234 do {
235 cTValue *tv = lj_tab_getint(t, i);
236 if (tv) {
237 copyTV(L, L->top++, tv);
238 } else {
239 setnilV(L->top++);
240 }
241 } while (i++ < e);
242 return n;
243}

245LJLIB_CF(select) LJLIB_REC(.)
246{
247 int32_t n = (int32_t)(L->top - L->base);
248 if (n >= 1 && tvisstr(L->base) && *strVdata(L->base) == '#') {
249 setintV(L->top-1, n-1);
250 return 1;
251 } else {
252 int32_t i = lj_lib_checkint(L, 1);
253 if (i < 0) i = n + i; else if (i > n) i = n;
254 if (i < 1)
255 lj_err_arg(L, 1, LJ_ERR_IDXRNG);
256 return n - i;
257 }
258}

262LJLIB_ASM(tonumber) LJLIB_REC(.)
263{
264 int32_t base = lj_lib_optint(L, 2, 10);
265 if (base == 10) {
266 TValue *o = lj_lib_checkany(L, 1);
267 if (lj_strscan_numberobj(o)) {
268 copyTV(L, L->base-1-LJ_FR2, o);
269 return FFH_RES(1);
270 }
271#if LJ_HASFFI
272 if (tviscdata(o)) {
273 CTState *cts = ctype_cts(L);
274 CType *ct = lj_ctype_rawref(cts, cdataV(o)->ctypeid);
275 if (ctype_isenum(ct->info)) ct = ctype_child(cts, ct);
276 if (ctype_isnum(ct->info) || ctype_iscomplex(ct->info)) {
277 if (LJ_DUALNUM && ctype_isinteger_or_bool(ct->info) &&
278 ct->size <= 4 && !(ct->size == 4 && (ct->info & CTF_UNSIGNED))) {
279 int32_t i;
280 lj_cconv_ct_tv(cts, ctype_get(cts, CTID_INT32), (uint8_t *)&i, o, 0);
281 setintV(L->base-1-LJ_FR2, i);
282 return FFH_RES(1);
283 }
284 lj_cconv_ct_tv(cts, ctype_get(cts, CTID_DOUBLE),
285 (uint8_t *)&(L->base-1-LJ_FR2)->n, o, 0);
286 return FFH_RES(1);
287 }
288 }
289#endif
290 } else {
291 const char *p = strdata(lj_lib_checkstr(L, 1));
292 char *ep;
293 unsigned int neg = 0;
294 unsigned long ul;
295 if (base < 2 || base > 36)
296 lj_err_arg(L, 2, LJ_ERR_BASERNG);
297 while (lj_char_isspace((unsigned char)(*p))) p++;
298 if (*p == '-') { p++; neg = 1; } else if (*p == '+') { p++; }
299 if (lj_char_isalnum((unsigned char)(*p))) {
300 ul = strtoul(p, &ep, base);
301 if (p != ep) {
302 while (lj_char_isspace((unsigned char)(*ep))) ep++;
303 if (*ep == '\0') {
304 if (LJ_DUALNUM && LJ_LIKELY(ul < 0x80000000u+neg)) {
305 if (neg) ul = (unsigned long)-(long)ul;
306 setintV(L->base-1-LJ_FR2, (int32_t)ul);
307 } else {
308 lua_Number n = (lua_Number)ul;
309 if (neg) n = -n;
310 setnumV(L->base-1-LJ_FR2, n);
311 }
312 return FFH_RES(1);
313 }
314 }
315 }
316 }
317 setnilV(L->base-1-LJ_FR2);
318 return FFH_RES(1);
319}

359static int load_aux(lua_State *L, int status, int envarg)
360{
361 if (status == LUA_OK) {
362 if (tvistab(L->base+envarg-1)) {
363 GCfunc *fn = funcV(L->top-1);
364 GCtab *t = tabV(L->base+envarg-1);
365 setgcref(fn->c.env, obj2gco(t));
366 lj_gc_objbarrier(L, fn, t);
367 }
368 return 1;
369 } else {
370 setnilV(L->top-2);
371 return 2;
372 }
373}

386static const char *reader_func(lua_State *L, void *ud, size_t *size)
387{
388 UNUSED(ud);
389 luaL_checkstack(L, 2, "too many nested functions");
390 copyTV(L, L->top++, L->base);
391 lua_call(L, 0, 1); /* Call user-supplied function. */
392 L->top--;
393 if (tvisnil(L->top)) {
394 *size = 0;
395 return NULL;
396 } else if (tvisstr(L->top) || tvisnumber(L->top)) {
397 copyTV(L, L->base+4, L->top); /* Anchor string in reserved stack slot. */
398 return lua_tolstring(L, 5, size);
399 } else {
400 lj_err_caller(L, LJ_ERR_RDRSTR);
401 return NULL;
402 }
403}

405LJLIB_CF(load)
406{
407 GCstr *name = lj_lib_optstr(L, 2);
408 GCstr *mode = lj_lib_optstr(L, 3);
409 int status;
410 if (L->base < L->top &&
411 (tvisstr(L->base) || tvisnumber(L->base) || tvisbuf(L->base))) {
412 const char *s;
413 MSize len;
414 if (tvisbuf(L->base)) {
415 SBufExt *sbx = bufV(L->base);
416 s = sbx->r;
417 len = sbufxlen(sbx);
418 if (!name) name = &G(L)->strempty; /* Buffers are not NUL-terminated. */
419 } else {
420 GCstr *str = lj_lib_checkstr(L, 1);
421 s = strdata(str);
422 len = str->len;
423 }
424 lua_settop(L, 4); /* Ensure env arg exists. */
425 status = luaL_loadbufferx(L, s, len, name ? strdata(name) : s,
426 mode ? strdata(mode) : NULL);
427 } else {
428 lj_lib_checkfunc(L, 1);
429 lua_settop(L, 5); /* Reserve a slot for the string from the reader. */
430 status = lua_loadx(L, reader_func, NULL, name ? strdata(name) : "=(load)",
431 mode ? strdata(mode) : NULL);
432 }
433 return load_aux(L, status, 4);
434}

460LJLIB_CF(collectgarbage)
461{
462 int opt = lj_lib_checkopt(L, 1, LUA_GCCOLLECT, /* ORDER LUA_GC* */
463 "\4stop\7restart\7collect\5count\1\377\4step\10setpause\12setstepmul\1\377\11isrunning");
464 int32_t data = lj_lib_optint(L, 2, 0);
465 if (opt == LUA_GCCOUNT) {
466 setnumV(L->top, (lua_Number)G(L)->gc.total/1024.0);
467 } else {
468 int res = lua_gc(L, opt, data);
469 if (opt == LUA_GCSTEP || opt == LUA_GCISRUNNING)
470 setboolV(L->top, res);
471 else
472 setintV(L->top, res);
473 }
474 L->top++;
475 return 1;
476}

480LJLIB_PUSH(top-2) /* Upvalue holds weak table. */
481LJLIB_CF(newproxy)
482{
483 lua_settop(L, 1);
484 lua_newuserdata(L, 0);
485 if (lua_toboolean(L, 1) == 0) { /* newproxy(): without metatable. */
486 return 1;
487 } else if (lua_isboolean(L, 1)) { /* newproxy(true): with metatable. */
488 lua_newtable(L);
489 lua_pushvalue(L, -1);
490 lua_pushboolean(L, 1);
491 lua_rawset(L, lua_upvalueindex(1)); /* Remember mt in weak table. */
492 } else { /* newproxy(proxy): inherit metatable. */
493 int validproxy = 0;
494 if (lua_getmetatable(L, 1)) {
495 lua_rawget(L, lua_upvalueindex(1));
496 validproxy = lua_toboolean(L, -1);
497 lua_pop(L, 1);
498 }
499 if (!validproxy)
500 lj_err_arg(L, 1, LJ_ERR_NOPROXY);
501 lua_getmetatable(L, 1);
502 }
503 lua_setmetatable(L, 2);
504 return 1;
505}

507LJLIB_PUSH("tostring")
508LJLIB_CF(print)
509{
510 ptrdiff_t i, nargs = L->top - L->base;
511 cTValue *tv = lj_tab_getstr(tabref(L->env), strV(lj_lib_upvalue(L, 1)));
512 int shortcut;
513 if (tv && !tvisnil(tv)) {
514 copyTV(L, L->top++, tv);
515 } else {
516 setstrV(L, L->top++, strV(lj_lib_upvalue(L, 1)));
517 lua_gettable(L, LUA_GLOBALSINDEX);
518 tv = L->top-1;
519 }
520 shortcut = (tvisfunc(tv) && funcV(tv)->c.ffid == FF_tostring) &&
521 !gcrefu(basemt_it(G(L), LJ_TNUMX));
522 for (i = 0; i < nargs; i++) {
523 cTValue *o = &L->base[i];
524 const char *str;
525 size_t size;
526 MSize len;
527 if (shortcut && (str = lj_strfmt_wstrnum(L, o, &len)) != NULL) {
528 size = len;
529 } else {
530 copyTV(L, L->top+1, o);
531 copyTV(L, L->top, L->top-1);
532 L->top += 2;
533 lua_call(L, 1, 1);
534 str = lua_tolstring(L, -1, &size);
535 if (!str)
536 lj_err_caller(L, LJ_ERR_PRTOSTR);
537 L->top--;
538 }
539 if (i)
540 putchar('\t');
541 fwrite(str, 1, size, stdout);
542 }
543 putchar('\n');
544 return 0;
545}

556LJLIB_CF(coroutine_status)
557{
558 const char *s;
559 lua_State *co;
560 if (!(L->top > L->base && tvisthread(L->base)))
561 lj_err_arg(L, 1, LJ_ERR_NOCORO);
562 co = threadV(L->base);
563 if (co == L) s = "running";
564 else if (co->status == LUA_YIELD) s = "suspended";
565 else if (co->status != LUA_OK) s = "dead";
566 else if (co->base > tvref(co->stack)+1+LJ_FR2) s = "normal";
567 else if (co->top == co->base) s = "dead";
568 else s = "suspended";
569 lua_pushstring(L, s);
570 return 1;
571}

608static int ffh_resume(lua_State *L, lua_State *co, int wrap)
609{
610 if (co->cframe != NULL || co->status > LUA_YIELD ||
611 (co->status == LUA_OK && co->top == co->base)) {
612 ErrMsg em = co->cframe ? LJ_ERR_CORUN : LJ_ERR_CODEAD;
613 if (wrap) lj_err_caller(L, em);
614 setboolV(L->base-1-LJ_FR2, 0);
615 setstrV(L, L->base-LJ_FR2, lj_err_str(L, em));
616 return FFH_RES(2);
617 }
618 lj_state_growstack(co, (MSize)(L->top - L->base));
619 return FFH_RETRY;
620}
